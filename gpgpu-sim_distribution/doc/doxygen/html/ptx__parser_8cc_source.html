305 // Returns padding that needs to be inserted ahead of address to make it aligned
306 // to min(size, maxalign)
307 /*
308  * @param address the address in bytes
309  * @param size the size of the memory to be allocated in bytes
310  * @param maximum alignment in bytes. i.e. if size is too big then align to this
311  * instead
312  */
313 int pad_address(new_addr_type address, unsigned size, unsigned maxalign) {
314  assert(size >= 0);
315  assert(maxalign > 0);
316  int alignto = maxalign;
317  if (size < maxalign && (size & (size - 1)) == 0) { // size is a power of 2
318  alignto = size;
319  }
320  return alignto ? ((alignto - (address % alignto)) % alignto) : 0;
321 }

323 void ptx_recognizer::add_identifier(const char *identifier, int array_dim,
324  unsigned array_ident) {
325  if (array_ident == ARRAY_IDENTIFIER) {
326  g_size *= array_dim;
327  }
328  if (g_func_decl && (g_func_info == NULL)) {
329  // return variable decl...
330  assert(g_add_identifier_cached__identifier == NULL);
331  g_add_identifier_cached__identifier = strdup(identifier);
332  g_add_identifier_cached__array_dim = array_dim;
333  g_add_identifier_cached__array_ident = array_ident;
334  return;
335  }
336  PTX_PARSE_DPRINTF("add_identifier \"%s\" (%u)", identifier, g_ident_add_uid);
337  g_ident_add_uid++;
338  type_info *type = g_var_type;
339  type_info_key ti = type->get_key();
340  int basic_type;
341  int regnum;
342  size_t num_bits;
343  unsigned addr_pad;
344  new_addr_type addr;
345  ti.type_decode(num_bits, basic_type);
346 
347  bool duplicates = check_for_duplicates(identifier);
348  if (duplicates) {
349  symbol *s = g_current_symbol_table->lookup(identifier);
350  g_last_symbol = s;
351  if (g_func_decl) return;
352  std::string msg = std::string(identifier) + " was declared previous at " +
353  s->decl_location() + " skipping new declaration";
354  printf("GPGPU-Sim PTX: Warning %s\n", msg.c_str());
355  return;
356  }
357 
358  assert(g_var_type != NULL);
359  switch (array_ident) {
360  case ARRAY_IDENTIFIER:
361  type = g_current_symbol_table->get_array_type(type, array_dim);
362  num_bits = array_dim * num_bits;
363  break;
364  case ARRAY_IDENTIFIER_NO_DIM:
365  type = g_current_symbol_table->get_array_type(type, (unsigned)-1);
366  num_bits = 0;
367  break;
368  default:
369  break;
370  }
371  g_last_symbol = g_current_symbol_table->add_variable(
372  identifier, type, num_bits / 8, gpgpu_ctx->g_filename,
373  ptx_get_lineno(scanner));
374  switch (ti.get_memory_space().get_type()) {
375  case reg_space: {
376  regnum = g_current_symbol_table->next_reg_num();
377  int arch_regnum = -1;
378  for (int d = 0; d < strlen(identifier); d++) {
379  if (isdigit(identifier[d])) {
380  sscanf(identifier + d, "%d", &arch_regnum);
381  break;
382  }
383  }
384  if (strcmp(identifier, "%sp") == 0) {
385  arch_regnum = 0;
386  }
387  g_last_symbol->set_regno(regnum, arch_regnum);
388  } break;
389  case shared_space:
390  printf("GPGPU-Sim PTX: allocating shared region for \"%s\" ", identifier);
391  fflush(stdout);
392  assert((num_bits % 8) == 0);
393  addr = g_current_symbol_table->get_shared_next();
394  addr_pad = pad_address(addr, num_bits / 8, 128);
395  printf("from 0x%llx to 0x%llx (shared memory space)\n", addr + addr_pad,
396  addr + addr_pad + num_bits / 8);
397  fflush(stdout);
398  g_last_symbol->set_address(addr + addr_pad);
399  g_current_symbol_table->alloc_shared(num_bits / 8 + addr_pad);
400  break;
401  case sstarr_space:
402  printf("GPGPU-Sim PTX: allocating sstarr region for \"%s\" ", identifier);
403  fflush(stdout);
404  assert((num_bits % 8) == 0);
405  addr = g_current_symbol_table->get_sstarr_next();
406  addr_pad = pad_address(addr, num_bits / 8, 128);
407  printf("from 0x%llx to 0x%llx (sstarr memory space)\n", addr + addr_pad,
408  addr + addr_pad + num_bits / 8);
409  fflush(stdout);
410  g_last_symbol->set_address(addr + addr_pad);
411  g_current_symbol_table->alloc_sstarr(num_bits / 8 + addr_pad);
412  break;
413  case const_space:
414  if (array_ident == ARRAY_IDENTIFIER_NO_DIM) {
415  printf(
416  "GPGPU-Sim PTX: deferring allocation of constant region for \"%s\" "
417  "(need size information)\n",
418  identifier);
419  } else {
420  printf("GPGPU-Sim PTX: allocating constant region for \"%s\" ",
421  identifier);
422  fflush(stdout);
423  assert((num_bits % 8) == 0);
424  addr = g_current_symbol_table->get_global_next();
425  addr_pad = pad_address(addr, num_bits / 8, 128);
426  printf("from 0x%llx to 0x%llx (global memory space) %u\n",
427  addr + addr_pad, addr + addr_pad + num_bits / 8,
428  g_const_alloc++);
429  fflush(stdout);
430  g_last_symbol->set_address(addr + addr_pad);
431  g_current_symbol_table->alloc_global(num_bits / 8 + addr_pad);
432  }
433  if (g_current_symbol_table == g_global_symbol_table) {
434  gpgpu_ctx->func_sim->g_constants.insert(identifier);
435  }
436  assert(g_current_symbol_table != NULL);
437  g_sym_name_to_symbol_table[identifier] = g_current_symbol_table;
438  break;
439  case global_space:
440  printf("GPGPU-Sim PTX: allocating global region for \"%s\" ", identifier);
441  fflush(stdout);
442  assert((num_bits % 8) == 0);
443  addr = g_current_symbol_table->get_global_next();
444  addr_pad = pad_address(addr, num_bits / 8, 128);
445  printf("from 0x%llx to 0x%llx (global memory space)\n", addr + addr_pad,
446  addr + addr_pad + num_bits / 8);
447  fflush(stdout);
448  g_last_symbol->set_address(addr + addr_pad);
449  g_current_symbol_table->alloc_global(num_bits / 8 + addr_pad);
450  gpgpu_ctx->func_sim->g_globals.insert(identifier);
451  assert(g_current_symbol_table != NULL);
452  g_sym_name_to_symbol_table[identifier] = g_current_symbol_table;
453  break;
454  case local_space:
455  if (g_func_info == NULL) {
456  printf("GPGPU-Sim PTX: allocating local region for \"%s\" ",
457  identifier);
458  fflush(stdout);
459  assert((num_bits % 8) == 0);
460  addr = g_current_symbol_table->get_local_next();
461  addr_pad = pad_address(addr, num_bits / 8, 128);
462  printf("from 0x%llx to 0x%llx (local memory space)\n", addr + addr_pad,
463  addr + addr_pad + num_bits / 8);
464  fflush(stdout);
465  g_last_symbol->set_address(addr + addr_pad);
466  g_current_symbol_table->alloc_local(num_bits / 8 + addr_pad);
467  } else {
468  printf(
469  "GPGPU-Sim PTX: allocating stack frame region for .local \"%s\" ",
470  identifier);
471  fflush(stdout);
472  assert((num_bits % 8) == 0);
473  addr = g_current_symbol_table->get_local_next();
474  addr_pad = pad_address(addr, num_bits / 8, 128);
475  printf("from 0x%llx to 0x%llx\n", addr + addr_pad,
476  addr + addr_pad + num_bits / 8);
477  fflush(stdout);
478  g_last_symbol->set_address(addr + addr_pad);
479  g_current_symbol_table->alloc_local(num_bits / 8 + addr_pad);
480  g_func_info->set_framesize(g_current_symbol_table->get_local_next());
481  }
482  break;
483  case tex_space:
484  printf("GPGPU-Sim PTX: encountered texture directive %s.\n", identifier);
485  break;
486  case param_space_local:
487  printf(
488  "GPGPU-Sim PTX: allocating stack frame region for .param \"%s\" from "
489  "0x%x to 0x%lx\n",
490  identifier, g_current_symbol_table->get_local_next(),
491  g_current_symbol_table->get_local_next() + num_bits / 8);
492  fflush(stdout);
493  assert((num_bits % 8) == 0);
494  g_last_symbol->set_address(g_current_symbol_table->get_local_next());
495  g_current_symbol_table->alloc_local(num_bits / 8);
496  g_func_info->set_framesize(g_current_symbol_table->get_local_next());
497  break;
498  case param_space_kernel:
499  break;
500  default:
501  abort();
502  break;
503  }
504 
505  assert(!ti.is_param_unclassified());
506  if (ti.is_param_kernel()) {
507  bool is_ptr = (g_ptr_spec != undefined_space);
508  g_func_info->add_param_name_type_size(g_entry_func_param_index, identifier,
509  ti.scalar_type(), num_bits, is_ptr,
510  g_ptr_spec);
511  g_entry_func_param_index++;
512  }
513 }
