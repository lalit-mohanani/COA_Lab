29 void cuda_device_runtime::gpgpusim_cuda_getParameterBufferV2(
30  const ptx_instruction *pI, ptx_thread_info *thread,
31  const function_info *target_func) {
32  DEV_RUNTIME_REPORT("Calling cudaGetParameterBufferV2");
33 
34  unsigned n_return = target_func->has_return();
35  assert(n_return);
36  unsigned n_args = target_func->num_args();
37  assert(n_args == 4);
38 
39  function_info *child_kernel_entry;
40  struct dim3 grid_dim, block_dim;
41  unsigned int shared_mem;
42 
43  for (unsigned arg = 0; arg < n_args; arg++) {
44  const operand_info &actual_param_op =
45  pI->operand_lookup(n_return + 1 + arg); // param#
46  const symbol *formal_param =
47  target_func->get_arg(arg); // cudaGetParameterBufferV2_param_#
48  unsigned size = formal_param->get_size_in_bytes();
49  assert(formal_param->is_param_local());
50  assert(actual_param_op.is_param_local());
51  addr_t from_addr = actual_param_op.get_symbol()->get_address();
52 
53  if (arg == 0) { // function_info* for the child kernel
54  unsigned long long buf;
55  assert(size == sizeof(function_info *));
56  thread->m_local_mem->read(from_addr, size, &buf);
57  child_kernel_entry = (function_info *)buf;
58  assert(child_kernel_entry);
59  DEV_RUNTIME_REPORT("child kernel name "
60  << child_kernel_entry->get_name());
61  } else if (arg == 1) { // dim3 grid_dim for the child kernel
62  assert(size == sizeof(struct dim3));
63  thread->m_local_mem->read(from_addr, size, &grid_dim);
64  DEV_RUNTIME_REPORT("grid (" << grid_dim.x << ", " << grid_dim.y << ", "
65  << grid_dim.z << ")");
66  } else if (arg == 2) { // dim3 block_dim for the child kernel
67  assert(size == sizeof(struct dim3));
68  thread->m_local_mem->read(from_addr, size, &block_dim);
69  DEV_RUNTIME_REPORT("block (" << block_dim.x << ", " << block_dim.y << ", "
70  << block_dim.z << ")");
71  } else if (arg == 3) { // unsigned int shared_mem
72  assert(size == sizeof(unsigned int));
73  thread->m_local_mem->read(from_addr, size, &shared_mem);
74  DEV_RUNTIME_REPORT("shared memory " << shared_mem);
75  }
76  }
77 
78  // get total child kernel argument size and malloc buffer in global memory
79  unsigned child_kernel_arg_size = child_kernel_entry->get_args_aligned_size();
80  void *param_buffer = thread->get_gpu()->gpu_malloc(child_kernel_arg_size);
81  g_total_param_size += ((child_kernel_arg_size + 255) / 256 * 256);
82  DEV_RUNTIME_REPORT("child kernel arg size total "
83  << child_kernel_arg_size
84  << ", parameter buffer allocated at " << param_buffer);
85  if (g_total_param_size > g_max_total_param_size)
86  g_max_total_param_size = g_total_param_size;
87 
88  // store param buffer address and launch config
89  device_launch_config_t device_launch_config(grid_dim, block_dim, shared_mem,
90  child_kernel_entry);
91  assert(g_cuda_device_launch_param_map.find(param_buffer) ==
92  g_cuda_device_launch_param_map.end());
93  g_cuda_device_launch_param_map[param_buffer] = device_launch_config;
94 
95  // copy the buffer address to retval0
96  const operand_info &actual_return_op = pI->operand_lookup(0); // retval0
97  const symbol *formal_return = target_func->get_return_var(); // void *
98  unsigned int return_size = formal_return->get_size_in_bytes();
99  DEV_RUNTIME_REPORT("cudaGetParameterBufferV2 return value has size of "
100  << return_size);
101  assert(actual_return_op.is_param_local());
102  assert(actual_return_op.get_symbol()->get_size_in_bytes() == return_size &&
103  return_size == sizeof(void *));
104  addr_t ret_param_addr = actual_return_op.get_symbol()->get_address();
105  thread->m_local_mem->write(ret_param_addr, return_size, &param_buffer, NULL,
106  NULL);
107 }

111 void cuda_device_runtime::gpgpusim_cuda_launchDeviceV2(
112  const ptx_instruction *pI, ptx_thread_info *thread,
113  const function_info *target_func) {
114  DEV_RUNTIME_REPORT("Calling cudaLaunchDeviceV2");
115 
116  unsigned n_return = target_func->has_return();
117  assert(n_return);
118  unsigned n_args = target_func->num_args();
119  assert(n_args == 2);
120 
121  kernel_info_t *device_grid = NULL;
122  function_info *device_kernel_entry = NULL;
123  void *parameter_buffer;
124  struct CUstream_st *child_stream;
125  device_launch_config_t config;
126  device_launch_operation_t device_launch_op;
127 
128  for (unsigned arg = 0; arg < n_args; arg++) {
129  const operand_info &actual_param_op =
130  pI->operand_lookup(n_return + 1 + arg); // param#
131  const symbol *formal_param =
132  target_func->get_arg(arg); // cudaLaunchDeviceV2_param_#
133  unsigned size = formal_param->get_size_in_bytes();
134  assert(formal_param->is_param_local());
135  assert(actual_param_op.is_param_local());
136  addr_t from_addr = actual_param_op.get_symbol()->get_address();
137 
138  if (arg == 0) { // paramter buffer for child kernel (in global memory)
139  // get parameter_buffer from the cudaLaunchDeviceV2_param0
140  assert(size == sizeof(void *));
141  thread->m_local_mem->read(from_addr, size, &parameter_buffer);
142  assert((size_t)parameter_buffer >= GLOBAL_HEAP_START);
143  DEV_RUNTIME_REPORT("Parameter buffer locating at global memory "
144  << parameter_buffer);
145 
146  // get child grid info through parameter_buffer address
147  assert(g_cuda_device_launch_param_map.find(parameter_buffer) !=
148  g_cuda_device_launch_param_map.end());
149  config = g_cuda_device_launch_param_map[parameter_buffer];
150  // device_grid = op.grid;
151  device_kernel_entry = config.entry;
152  DEV_RUNTIME_REPORT("find device kernel "
153  << device_kernel_entry->get_name());
154 
155  // PDOM analysis is done for Parent kernel but not for child kernel.
156  if (device_kernel_entry->is_pdom_set()) {
157  printf("GPGPU-Sim PTX: PDOM analysis already done for %s \n",
158  device_kernel_entry->get_name().c_str());
159  } else {
160  printf("GPGPU-Sim PTX: finding reconvergence points for \'%s\'...\n",
161  device_kernel_entry->get_name().c_str());
162  /*
163  * Some of the instructions like printf() gives the gpgpusim the wrong
164  * impression that it is a function call. As printf() doesnt have a body
165  * like functions do, doing pdom analysis for printf() causes a crash.
166  */
167  if (device_kernel_entry->get_function_size() > 0)
168  device_kernel_entry->do_pdom();
169  device_kernel_entry->set_pdom();
170  }
171 
172  // copy data in parameter_buffer to device kernel param memory
173  unsigned device_kernel_arg_size =
174  device_kernel_entry->get_args_aligned_size();
175  DEV_RUNTIME_REPORT("device_kernel_arg_size " << device_kernel_arg_size);
176  memory_space *device_kernel_param_mem;
177 
178  // create child kernel_info_t and index it with parameter_buffer address
179  gpgpu_t *gpu = thread->get_gpu();
180  device_grid = new kernel_info_t(
181  config.grid_dim, config.block_dim, device_kernel_entry,
182  gpu->getNameArrayMapping(), gpu->getNameInfoMapping());
183  device_grid->launch_cycle = gpu->gpu_sim_cycle + gpu->gpu_tot_sim_cycle;
184  kernel_info_t &parent_grid = thread->get_kernel();
185  DEV_RUNTIME_REPORT(
186  "child kernel launched by "
187  << parent_grid.name() << ", cta (" << thread->get_ctaid().x << ", "
188  << thread->get_ctaid().y << ", " << thread->get_ctaid().z
189  << "), thread (" << thread->get_tid().x << ", " << thread->get_tid().y
190  << ", " << thread->get_tid().z << ")");
191  device_grid->set_parent(&parent_grid, thread->get_ctaid(),
192  thread->get_tid());
193  device_launch_op = device_launch_operation_t(device_grid, NULL);
194  device_kernel_param_mem = device_grid->get_param_memory(); // kernel
195  // param
196  size_t param_start_address = 0;
197  // copy in word
198  for (unsigned n = 0; n < device_kernel_arg_size; n += 4) {
199  unsigned int oneword;
200  thread->get_gpu()->get_global_memory()->read(
201  (size_t)parameter_buffer + n, 4, &oneword);
202  device_kernel_param_mem->write(param_start_address + n, 4, &oneword,
203  NULL, NULL);
204  }
205  } else if (arg == 1) { // cudaStream for the child kernel
206 
207  assert(size == sizeof(cudaStream_t));
208  thread->m_local_mem->read(from_addr, size, &child_stream);
209 
210  kernel_info_t &parent_kernel = thread->get_kernel();
211  if (child_stream == 0) { // default stream on device for current CTA
212  child_stream =
213  parent_kernel.get_default_stream_cta(thread->get_ctaid());
214  DEV_RUNTIME_REPORT("launching child kernel "
215  << device_grid->get_uid()
216  << " to default stream of the cta "
217  << child_stream->get_uid() << ": " << child_stream);
218  } else {
219  assert(parent_kernel.cta_has_stream(thread->get_ctaid(), child_stream));
220  DEV_RUNTIME_REPORT("launching child kernel "
221  << device_grid->get_uid() << " to stream "
222  << child_stream->get_uid() << ": " << child_stream);
223  }
224 
225  device_launch_op.stream = child_stream;
226  }
227  }
228 
229  // launch child kernel
230  g_cuda_device_launch_op.push_back(device_launch_op);
231  g_cuda_device_launch_param_map.erase(parameter_buffer);
232 
233  // set retval0
234  const operand_info &actual_return_op = pI->operand_lookup(0); // retval0
235  const symbol *formal_return = target_func->get_return_var(); // cudaError_t
236  unsigned int return_size = formal_return->get_size_in_bytes();
237  DEV_RUNTIME_REPORT("cudaLaunchDeviceV2 return value has size of "
238  << return_size);
239  assert(actual_return_op.is_param_local());
240  assert(actual_return_op.get_symbol()->get_size_in_bytes() == return_size &&
241  return_size == sizeof(cudaError_t));
242  cudaError_t error = cudaSuccess;
243  addr_t ret_param_addr = actual_return_op.get_symbol()->get_address();
244  thread->m_local_mem->write(ret_param_addr, return_size, &error, NULL, NULL);
245 }

258  assert(n_args == 2);
259 
260  size_t generic_pStream_addr;
261  addr_t pStream_addr;
262  unsigned int flags;
263  for (unsigned arg = 0; arg < n_args; arg++) {
264  const operand_info &actual_param_op =
265  pI->operand_lookup(n_return + 1 + arg); // param#
266  const symbol *formal_param =

272

275  thread->m_local_mem->read(from_addr, size, &generic_pStream_addr);
276 
277  // pStream should be non-zero address in local memory
278  pStream_addr = generic_to_local(
279  thread->get_hw_sid(), thread->get_hw_tid(), generic_pStream_addr);
280

282  } else if (arg ==

285  thread->m_local_mem->read(from_addr, size, &flags);
286  assert(flags == cudaStreamNonBlocking);
287  }
288  }
289 
290  // create stream and write back to param0
291  CUstream_st *stream =
292  thread->get_kernel().create_stream_cta(thread->get_ctaid());
293  DEV_RUNTIME_REPORT("Create stream " << stream->get_uid() << ": " << stream);
294  thread->m_local_mem->write(pStream_addr, sizeof(cudaStream_t), &stream, NULL,
295  NULL);
296 
297  // set retval0
298  const operand_info &actual_return_op = pI->operand_lookup(0); // retval0
299  const symbol *formal_return = target_func->get_return_var(); // cudaError_t
300  unsigned int return_size = formal_return->get_size_in_bytes();
301  DEV_RUNTIME_REPORT("cudaStreamCreateWithFlags return value has size of "
302  << return_size);
303  assert(actual_return_op.is_param_local());
304  assert(actual_return_op.get_symbol()->get_size_in_bytes() == return_size &&
305  return_size == sizeof(cudaError_t));
306  cudaError_t error = cudaSuccess;
307  addr_t ret_param_addr = actual_return_op.get_symbol()->get_address();
308  thread->m_local_mem->write(ret_param_addr, return_size, &error, NULL, NULL);
309 }
