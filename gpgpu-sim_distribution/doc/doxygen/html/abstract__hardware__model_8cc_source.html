281 void warp_inst_t::generate_mem_accesses() {
282  if (empty() || op == MEMORY_BARRIER_OP || m_mem_accesses_created) return;
283  if (!((op == LOAD_OP) || (op == TENSOR_CORE_LOAD_OP) || (op == STORE_OP) ||
284  (op == TENSOR_CORE_STORE_OP)))
285  return;
286  if (m_warp_active_mask.count() == 0) return; // predicated off
287 
288  const size_t starting_queue_size = m_accessq.size();
289 
290  assert(is_load() || is_store());
291  assert(m_per_scalar_thread_valid); // need address information per thread
292 
293  bool is_write = is_store();
294 
295  mem_access_type access_type;
296  switch (space.get_type()) {
297  case const_space:
298  case param_space_kernel:
299  access_type = CONST_ACC_R;
300  break;
301  case tex_space:
302  access_type = TEXTURE_ACC_R;
303  break;
304  case global_space:
305  access_type = is_write ? GLOBAL_ACC_W : GLOBAL_ACC_R;
306  break;
307  case local_space:
308  case param_space_local:
309  access_type = is_write ? LOCAL_ACC_W : LOCAL_ACC_R;
310  break;
311  case shared_space:
312  break;
313  case sstarr_space:
314  break;
315  default:
316  assert(0);
317  break;
318  }
319 
320  // Calculate memory accesses generated by this warp
321  new_addr_type cache_block_size = 0; // in bytes
322 
323  switch (space.get_type()) {
324  case shared_space:
325  case sstarr_space: {
326  unsigned subwarp_size = m_config->warp_size / m_config->mem_warp_parts;
327  unsigned total_accesses = 0;
328  for (unsigned subwarp = 0; subwarp < m_config->mem_warp_parts;
329  subwarp++) {
330  // data structures used per part warp
331  std::map<unsigned, std::map<new_addr_type, unsigned> >
332  bank_accs; // bank -> word address -> access count
333 
334  // step 1: compute accesses to words in banks
335  for (unsigned thread = subwarp * subwarp_size;
336  thread < (subwarp + 1) * subwarp_size; thread++) {
337  if (!active(thread)) continue;
338  new_addr_type addr = m_per_scalar_thread[thread].memreqaddr[0];
339  // FIXME: deferred allocation of shared memory should not accumulate
340  // across kernel launches assert( addr < m_config->gpgpu_shmem_size );
341  unsigned bank = m_config->shmem_bank_func(addr);
342  new_addr_type word =
343  line_size_based_tag_func(addr, m_config->WORD_SIZE);
344  bank_accs[bank][word]++;
345  }
346 
347  if (m_config->shmem_limited_broadcast) {
348  // step 2: look for and select a broadcast bank/word if one occurs
349  bool broadcast_detected = false;
350  new_addr_type broadcast_word = (new_addr_type)-1;
351  unsigned broadcast_bank = (unsigned)-1;
352  std::map<unsigned, std::map<new_addr_type, unsigned> >::iterator b;
353  for (b = bank_accs.begin(); b != bank_accs.end(); b++) {
354  unsigned bank = b->first;
355  std::map<new_addr_type, unsigned> &access_set = b->second;
356  std::map<new_addr_type, unsigned>::iterator w;
357  for (w = access_set.begin(); w != access_set.end(); ++w) {
358  if (w->second > 1) {
359  // found a broadcast
360  broadcast_detected = true;
361  broadcast_bank = bank;
362  broadcast_word = w->first;
363  break;
364  }
365  }
366  if (broadcast_detected) break;
367  }
368 
369  // step 3: figure out max bank accesses performed, taking account of
370  // broadcast case
371  unsigned max_bank_accesses = 0;
372  for (b = bank_accs.begin(); b != bank_accs.end(); b++) {
373  unsigned bank_accesses = 0;
374  std::map<new_addr_type, unsigned> &access_set = b->second;
375  std::map<new_addr_type, unsigned>::iterator w;
376  for (w = access_set.begin(); w != access_set.end(); ++w)
377  bank_accesses += w->second;
378  if (broadcast_detected && broadcast_bank == b->first) {
379  for (w = access_set.begin(); w != access_set.end(); ++w) {
380  if (w->first == broadcast_word) {
381  unsigned n = w->second;
382  assert(n > 1); // or this wasn't a broadcast
383  assert(bank_accesses >= (n - 1));
384  bank_accesses -= (n - 1);
385  break;
386  }
387  }
388  }
389  if (bank_accesses > max_bank_accesses)
390  max_bank_accesses = bank_accesses;
391  }
392 
393  // step 4: accumulate
394  total_accesses += max_bank_accesses;
395  } else {
396  // step 2: look for the bank with the maximum number of access to
397  // different words
398  unsigned max_bank_accesses = 0;
399  std::map<unsigned, std::map<new_addr_type, unsigned> >::iterator b;
400  for (b = bank_accs.begin(); b != bank_accs.end(); b++) {
401  max_bank_accesses =
402  std::max(max_bank_accesses, (unsigned)b->second.size());
403  }
404 
405  // step 3: accumulate
406  total_accesses += max_bank_accesses;
407  }
408  }
409  assert(total_accesses > 0 && total_accesses <= m_config->warp_size);
410  cycles = total_accesses; // shared memory conflicts modeled as larger
411  // initiation interval
412  m_config->gpgpu_ctx->stats->ptx_file_line_stats_add_smem_bank_conflict(
413  pc, total_accesses);
414  break;
415  }
416 
417  case tex_space:
418  cache_block_size = m_config->gpgpu_cache_texl1_linesize;
419  break;
420  case const_space:
421  case param_space_kernel:
422  cache_block_size = m_config->gpgpu_cache_constl1_linesize;
423  break;
424 
425  case global_space:
426  case local_space:
427  case param_space_local:
428  if (m_config->gpgpu_coalesce_arch >= 13) {
429  if (isatomic())
430  memory_coalescing_arch_atomic(is_write, access_type);
431  else
432  memory_coalescing_arch(is_write, access_type);
433  } else
434  abort();
435 
436  break;
437 
438  default:
439  abort();
440  }
441 
442  if (cache_block_size) {
443  assert(m_accessq.empty());
444  mem_access_byte_mask_t byte_mask;
445  std::map<new_addr_type, active_mask_t>
446  accesses; // block address -> set of thread offsets in warp
447  std::map<new_addr_type, active_mask_t>::iterator a;
448  for (unsigned thread = 0; thread < m_config->warp_size; thread++) {
449  if (!active(thread)) continue;
450  new_addr_type addr = m_per_scalar_thread[thread].memreqaddr[0];
451  unsigned block_address = line_size_based_tag_func(addr, cache_block_size);
452  accesses[block_address].set(thread);
453  unsigned idx = addr - block_address;
454  for (unsigned i = 0; i < data_size; i++) byte_mask.set(idx + i);
455  }
456  for (a = accesses.begin(); a != accesses.end(); ++a)
457  m_accessq.push_back(mem_access_t(
458  access_type, a->first, cache_block_size, is_write, a->second,
459  byte_mask, mem_access_sector_mask_t(), m_config->gpgpu_ctx));
460  }
461 
462  if (space.get_type() == global_space) {
463  m_config->gpgpu_ctx->stats->ptx_file_line_stats_add_uncoalesced_gmem(
464  pc, m_accessq.size() - starting_queue_size);
465  }
466  m_mem_accesses_created = true;
467 }

469 void warp_inst_t::memory_coalescing_arch(bool is_write,
470  mem_access_type access_type) {
471  // see the CUDA manual where it discusses coalescing rules before reading this
472  unsigned segment_size = 0;
473  unsigned warp_parts = m_config->mem_warp_parts;
474  bool sector_segment_size = false;
475 
476  if (m_config->gpgpu_coalesce_arch >= 20 &&
477  m_config->gpgpu_coalesce_arch < 39) {
478  // Fermi and Kepler, L1 is normal and L2 is sector
479  if (m_config->gmem_skip_L1D || cache_op == CACHE_GLOBAL)
480  sector_segment_size = true;
481  else
482  sector_segment_size = false;
483  } else if (m_config->gpgpu_coalesce_arch >= 40) {
484  // Maxwell, Pascal and Volta, L1 and L2 are sectors
485  // all requests should be 32 bytes
486  sector_segment_size = true;
487  }
488 
489  switch (data_size) {
490  case 1:
491  segment_size = 32;
492  break;
493  case 2:
494  segment_size = sector_segment_size ? 32 : 64;
495  break;
496  case 4:
497  case 8:
498  case 16:
499  segment_size = sector_segment_size ? 32 : 128;
500  break;
501  }
502  unsigned subwarp_size = m_config->warp_size / warp_parts;
503 
504  for (unsigned subwarp = 0; subwarp < warp_parts; subwarp++) {
505  std::map<new_addr_type, transaction_info> subwarp_transactions;
506 
507  // step 1: find all transactions generated by this subwarp
508  for (unsigned thread = subwarp * subwarp_size;
509  thread < subwarp_size * (subwarp + 1); thread++) {
510  if (!active(thread)) continue;
511 
512  unsigned data_size_coales = data_size;
513  unsigned num_accesses = 1;
514 
515  if (space.get_type() == local_space ||
516  space.get_type() == param_space_local) {
517  // Local memory accesses >4B were split into 4B chunks
518  if (data_size >= 4) {
519  data_size_coales = 4;
520  num_accesses = data_size / 4;
521  }
522  // Otherwise keep the same data_size for sub-4B access to local memory
523  }
524 
525  assert(num_accesses <= MAX_ACCESSES_PER_INSN_PER_THREAD);
526 
527  // for(unsigned access=0; access<num_accesses; access++) {
528  for (unsigned access = 0;
529  (access < MAX_ACCESSES_PER_INSN_PER_THREAD) &&
530  (m_per_scalar_thread[thread].memreqaddr[access] != 0);
531  access++) {
532  new_addr_type addr = m_per_scalar_thread[thread].memreqaddr[access];
533  unsigned block_address = line_size_based_tag_func(addr, segment_size);
534  unsigned chunk =
535  (addr & 127) / 32; // which 32-byte chunk within in a 128-byte
536  // chunk does this thread access?
537  transaction_info &info = subwarp_transactions[block_address];
538 
539  // can only write to one segment
540  // it seems like in trace driven, a thread can write to more than one
541  // segment assert(block_address ==
542  // line_size_based_tag_func(addr+data_size_coales-1,segment_size));
543 
544  info.chunks.set(chunk);
545  info.active.set(thread);
546  unsigned idx = (addr & 127);
547  for (unsigned i = 0; i < data_size_coales; i++)
548  if ((idx + i) < MAX_MEMORY_ACCESS_SIZE) info.bytes.set(idx + i);
549 
550  // it seems like in trace driven, a thread can write to more than one
551  // segment handle this special case
552  if (block_address != line_size_based_tag_func(
553  addr + data_size_coales - 1, segment_size)) {
554  addr = addr + data_size_coales - 1;
555  unsigned block_address = line_size_based_tag_func(addr, segment_size);
556  unsigned chunk = (addr & 127) / 32;
557  transaction_info &info = subwarp_transactions[block_address];
558  info.chunks.set(chunk);
559  info.active.set(thread);
560  unsigned idx = (addr & 127);
561  for (unsigned i = 0; i < data_size_coales; i++)
562  if ((idx + i) < MAX_MEMORY_ACCESS_SIZE) info.bytes.set(idx + i);
563  }
564  }
565  }
566 
567  // step 2: reduce each transaction size, if possible
568  std::map<new_addr_type, transaction_info>::iterator t;
569  for (t = subwarp_transactions.begin(); t != subwarp_transactions.end();
570  t++) {
571  new_addr_type addr = t->first;

580 void warp_inst_t::memory_coalescing_arch_atomic(bool is_write,
581  mem_access_type access_type) {
582  assert(space.get_type() ==
583  global_space); // Atomics allowed only for global memory
584 
585  // see the CUDA manual where it discusses coalescing rules before reading this
586  unsigned segment_size = 0;
587  unsigned warp_parts = m_config->mem_warp_parts;
588  bool sector_segment_size = false;
589 
590  if (m_config->gpgpu_coalesce_arch >= 20 &&
591  m_config->gpgpu_coalesce_arch < 39) {
592  // Fermi and Kepler, L1 is normal and L2 is sector
593  if (m_config->gmem_skip_L1D || cache_op == CACHE_GLOBAL)
594  sector_segment_size = true;
595  else
596  sector_segment_size = false;
597  } else if (m_config->gpgpu_coalesce_arch >= 40) {
598  // Maxwell, Pascal and Volta, L1 and L2 are sectors
599  // all requests should be 32 bytes
600  sector_segment_size = true;
601  }
602 
603  switch (data_size) {
604  case 1:
605  segment_size = 32;
606  break;
607  case 2:
608  segment_size = sector_segment_size ? 32 : 64;
609  break;
610  case 4:
611  case 8:
612  case 16:
613  segment_size = sector_segment_size ? 32 : 128;
614  break;
615  }
616  unsigned subwarp_size = m_config->warp_size / warp_parts;
617 
618  for (unsigned subwarp = 0; subwarp < warp_parts; subwarp++) {
619  std::map<new_addr_type, std::list<transaction_info> >
620  subwarp_transactions; // each block addr maps to a list of transactions
621 
622  // step 1: find all transactions generated by this subwarp
623  for (unsigned thread = subwarp * subwarp_size;
624  thread < subwarp_size * (subwarp + 1); thread++) {
625  if (!active(thread)) continue;
626 
627  new_addr_type addr = m_per_scalar_thread[thread].memreqaddr[0];
628  unsigned block_address = line_size_based_tag_func(addr, segment_size);
629  unsigned chunk =
630  (addr & 127) / 32; // which 32-byte chunk within in a 128-byte chunk
631  // does this thread access?
632 
633  // can only write to one segment
634  assert(block_address ==
635  line_size_based_tag_func(addr + data_size - 1, segment_size));
636 
637  // Find a transaction that does not conflict with this thread's accesses
638  bool new_transaction = true;
639  std::list<transaction_info>::iterator it;
640  transaction_info *info;
641  for (it = subwarp_transactions[block_address].begin();
642  it != subwarp_transactions[block_address].end(); it++) {
643  unsigned idx = (addr & 127);
644  if (not it->test_bytes(idx, idx + data_size - 1)) {
645  new_transaction = false;
646  info = &(*it);
647  break;
648  }
649  }
650  if (new_transaction) {
651  // Need a new transaction
652  subwarp_transactions[block_address].push_back(transaction_info());
653  info = &subwarp_transactions[block_address].back();
654  }
655  assert(info);
656 
657  info->chunks.set(chunk);
658  info->active.set(thread);
659  unsigned idx = (addr & 127);
660  for (unsigned i = 0; i < data_size; i++) {
661  assert(!info->bytes.test(idx + i));
662  info->bytes.set(idx + i);
663  }
664  }
665 
666  // step 2: reduce each transaction size, if possible
667  std::map<new_addr_type, std::list<transaction_info> >::iterator t_list;
668  for (t_list = subwarp_transactions.begin();
669  t_list != subwarp_transactions.end(); t_list++) {
670  // For each block addr
671  new_addr_type addr = t_list->first;
672  const std::list<transaction_info> &transaction_list = t_list->second;
673 
674  std::list<transaction_info>::const_iterator t;
675  for (t = transaction_list.begin(); t != transaction_list.end(); t++) {
676  // For each transaction
677  const transaction_info &info = *t;
678  memory_coalescing_arch_reduce_and_send(is_write, access_type, info,
679  addr, segment_size);
680  }
681  }
682  }
683 }

685 void warp_inst_t::memory_coalescing_arch_reduce_and_send(
686  bool is_write, mem_access_type access_type, const transaction_info &info,
687  new_addr_type addr, unsigned segment_size) {
688  assert((addr & (segment_size - 1)) == 0);
689 
690  const std::bitset<4> &q = info.chunks;
691  assert(q.count() >= 1);
692  std::bitset<2> h; // halves (used to check if 64 byte segment can be
693  // compressed into a single 32 byte segment)
694 
695  unsigned size = segment_size;
696  if (segment_size == 128) {
697  bool lower_half_used = q[0] || q[1];
698  bool upper_half_used = q[2] || q[3];
699  if (lower_half_used && !upper_half_used) {
700  // only lower 64 bytes used
701  size = 64;
702  if (q[0]) h.set(0);
703  if (q[1]) h.set(1);
704  } else if ((!lower_half_used) && upper_half_used) {
705  // only upper 64 bytes used
706  addr = addr + 64;
707  size = 64;
708  if (q[2]) h.set(0);
709  if (q[3]) h.set(1);
710  } else {
711  assert(lower_half_used && upper_half_used);
712  }
713  } else if (segment_size == 64) {
714  // need to set halves
715  if ((addr % 128) == 0) {
716  if (q[0]) h.set(0);
717  if (q[1]) h.set(1);
718  } else {
719  assert((addr % 128) == 64);
720  if (q[2]) h.set(0);
721  if (q[3]) h.set(1);
722  }
723  }
724  if (size == 64) {
725  bool lower_half_used = h[0];
726  bool upper_half_used = h[1];
727  if (lower_half_used && !upper_half_used) {
728  size = 32;
729  } else if ((!lower_half_used) && upper_half_used) {
730  addr = addr + 32;
731  size = 32;
732  } else {
733  assert(lower_half_used && upper_half_used);
734  }
735  }
736  m_accessq.push_back(mem_access_t(access_type, addr, size, is_write,
737  info.active, info.bytes, info.chunks,
738  m_config->gpgpu_ctx));
739 }

1033 void simt_stack::update(simt_mask_t &thread_done, addr_vector_t &next_pc,
1034  address_type recvg_pc, op_type next_inst_op,
1035  unsigned next_inst_size, address_type next_inst_pc) {
1036  assert(m_stack.size() > 0);
1037 
1038  assert(next_pc.size() == m_warp_size);
1039 
1040  simt_mask_t top_active_mask = m_stack.back().m_active_mask;
1041  address_type top_recvg_pc = m_stack.back().m_recvg_pc;
1042  address_type top_pc =
1043  m_stack.back().m_pc; // the pc of the instruction just executed
1044  stack_entry_type top_type = m_stack.back().m_type;
1045  assert(top_pc == next_inst_pc);
1046  assert(top_active_mask.any());
1047 
1048  const address_type null_pc = -1;
1049  bool warp_diverged = false;
1050  address_type new_recvg_pc = null_pc;
1051  unsigned num_divergent_paths = 0;
1052 
1053  std::map<address_type, simt_mask_t> divergent_paths;
1054  while (top_active_mask.any()) {
1055  // extract a group of threads with the same next PC among the active threads
1056  // in the warp
1057  address_type tmp_next_pc = null_pc;
1058  simt_mask_t tmp_active_mask;
1059  for (int i = m_warp_size - 1; i >= 0; i--) {
1060  if (top_active_mask.test(i)) { // is this thread active?
1061  if (thread_done.test(i)) {
1062  top_active_mask.reset(i); // remove completed thread from active mask
1063  } else if (tmp_next_pc == null_pc) {
1064  tmp_next_pc = next_pc[i];
1065  tmp_active_mask.set(i);
1066  top_active_mask.reset(i);
1067  } else if (tmp_next_pc == next_pc[i]) {
1068  tmp_active_mask.set(i);
1069  top_active_mask.reset(i);
1070  }
1071  }
1072  }
1073 
1074  if (tmp_next_pc == null_pc) {
1075  assert(!top_active_mask.any()); // all threads done
1076  continue;
1077  }
1078 
1079  divergent_paths[tmp_next_pc] = tmp_active_mask;
1080  num_divergent_paths++;
1081  }
1082 
1083  address_type not_taken_pc = next_inst_pc + next_inst_size;
1084  assert(num_divergent_paths <= 2);
1085  for (unsigned i = 0; i < num_divergent_paths; i++) {
1086  address_type tmp_next_pc = null_pc;
1087  simt_mask_t tmp_active_mask;
1088  tmp_active_mask.reset();
1089  if (divergent_paths.find(not_taken_pc) != divergent_paths.end()) {
1090  assert(i == 0);
1091  tmp_next_pc = not_taken_pc;
1092  tmp_active_mask = divergent_paths[tmp_next_pc];
1093  divergent_paths.erase(tmp_next_pc);
1094  } else {
1095  std::map<address_type, simt_mask_t>::iterator it =
1096  divergent_paths.begin();
1097  tmp_next_pc = it->first;
1098  tmp_active_mask = divergent_paths[tmp_next_pc];
1099  divergent_paths.erase(tmp_next_pc);
1100  }
1101 
1102  // HANDLE THE SPECIAL CASES FIRST
1103  if (next_inst_op == CALL_OPS) {
1104  // Since call is not a divergent instruction, all threads should have
1105  // executed a call instruction
1106  assert(num_divergent_paths == 1);
1107 
1108  simt_stack_entry new_stack_entry;
1109  new_stack_entry.m_pc = tmp_next_pc;
1110  new_stack_entry.m_active_mask = tmp_active_mask;
1111  new_stack_entry.m_branch_div_cycle =
1112  m_gpu->gpu_sim_cycle + m_gpu->gpu_tot_sim_cycle;
1113  new_stack_entry.m_type = STACK_ENTRY_TYPE_CALL;
1114  m_stack.push_back(new_stack_entry);
1115  return;
1116  } else if (next_inst_op == RET_OPS && top_type == STACK_ENTRY_TYPE_CALL) {
1117  // pop the CALL Entry
1118  assert(num_divergent_paths == 1);
1119  m_stack.pop_back();
1120 
1121  assert(m_stack.size() > 0);
