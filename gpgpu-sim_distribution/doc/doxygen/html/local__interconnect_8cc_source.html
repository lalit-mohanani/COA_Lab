123 void xbar_router::RR_Advance() {
124  bool active = false;
125  vector<bool> issued(total_nodes, false);
126  unsigned conflict_sub = 0;
127  unsigned reqs = 0;
128 
129  for (unsigned i = 0; i < total_nodes; ++i) {
130  unsigned node_id = (i + next_node_id) % total_nodes;
131 
132  if (!in_buffers[node_id].empty()) {
133  active = true;
134  Packet _packet = in_buffers[node_id].front();
135  // ensure that the outbuffer has space and not issued before in this cycle
136  if (Has_Buffer_Out(_packet.output_deviceID, 1)) {
137  if (!issued[_packet.output_deviceID]) {
138  out_buffers[_packet.output_deviceID].push(_packet);
139  in_buffers[node_id].pop();
140  issued[_packet.output_deviceID] = true;
141  reqs++;
142  } else
143  conflict_sub++;
144  } else {
145  out_buffer_full++;
146 
147  if (issued[_packet.output_deviceID]) conflict_sub++;
148  }
149  }
150  }
151 
152  next_node_id = (++next_node_id % total_nodes);
153 
154  conflicts += conflict_sub;
155  if (active) {
156  conflicts_util += conflict_sub;
157  cycles_util++;
158  reqs_util += reqs;
159  }
160 
161  if (verbose) {
162  printf("%d : cycle %d : conflicts = %d\n", m_id, cycles, conflict_sub);
163  printf("%d : cycle %d : passing reqs = %d\n", m_id, cycles, reqs);
164  }
165 
166  // collect some stats about buffer util
167  for (unsigned i = 0; i < total_nodes; ++i) {
168  in_buffer_util += in_buffers[i].size();
169  out_buffer_util += out_buffers[i].size();
170  }
171 
172  cycles++;
173 }

175 // iSLIP algorithm
176 // McKeown, Nick. "The iSLIP scheduling algorithm for input-queued switches."
177 // IEEE/ACM transactions on networking 2 (1999): 188-201.
178 // https://www.cs.rutgers.edu/~sn624/552-F18/papers/islip.pdf
179 void xbar_router::iSLIP_Advance() {
180  vector<unsigned> node_tmp;
181  bool active = false;
182 
183  unsigned conflict_sub = 0;
184  unsigned reqs = 0;
185 
186  // calcaulte how many conflicts are there for stats
187  for (unsigned i = 0; i < total_nodes; ++i) {
188  if (!in_buffers[i].empty()) {
189  Packet _packet_tmp = in_buffers[i].front();
190  if (!node_tmp.empty()) {
191  if (std::find(node_tmp.begin(), node_tmp.end(),
192  _packet_tmp.output_deviceID) != node_tmp.end()) {
193  conflict_sub++;
194  } else
195  node_tmp.push_back(_packet_tmp.output_deviceID);
196  } else {
197  node_tmp.push_back(_packet_tmp.output_deviceID);
198  }
199  active = true;
200  }
201  }
202 
203  conflicts += conflict_sub;
204  if (active) {
205  conflicts_util += conflict_sub;
206  cycles_util++;
207  }
208  // do iSLIP
209  for (unsigned i = 0; i < total_nodes; ++i) {
210  if (Has_Buffer_Out(i, 1)) {
211  for (unsigned j = 0; j < total_nodes; ++j) {
212  unsigned node_id = (j + next_node[i]) % total_nodes;
213 
214  if (!in_buffers[node_id].empty()) {
215  Packet _packet = in_buffers[node_id].front();
216  if (_packet.output_deviceID == i) {
217  out_buffers[_packet.output_deviceID].push(_packet);
218  in_buffers[node_id].pop();
219  if (verbose)
220  printf("%d : cycle %d : send req from %d to %d\n", m_id, cycles,
221  node_id, i - _n_shader);
222  if (grant_cycles_count == 1)
223  next_node[i] = (++node_id % total_nodes);
224  if (verbose) {
225  for (unsigned k = j + 1; k < total_nodes; ++k) {
226  unsigned node_id2 = (k + next_node[i]) % total_nodes;
227  if (!in_buffers[node_id2].empty()) {
228  Packet _packet2 = in_buffers[node_id2].front();
229 
230  if (_packet2.output_deviceID == i)
231  printf("%d : cycle %d : cannot send req from %d to %d\n",
232  m_id, cycles, node_id2, i - _n_shader);
233  }
234  }
235  }
236 
237  reqs++;
238  break;
239  }
240  }
241  }
242  } else
243  out_buffer_full++;
244  }
245 
246  if (active) {
247  reqs_util += reqs;
248  }
249 
250  if (verbose)
251  printf("%d : cycle %d : grant_cycles = %d\n", m_id, cycles, grant_cycles);
252 
253  if (active && grant_cycles_count == 1)
254  grant_cycles_count = grant_cycles;
255  else if (active)
256  grant_cycles_count--;
257 
258  if (verbose) {
259  printf("%d : cycle %d : conflicts = %d\n", m_id, cycles, conflict_sub);
260  printf("%d : cycle %d : passing reqs = %d\n", m_id, cycles, reqs);
261  }
262 
263  // collect some stats about buffer util
264  for (unsigned i = 0; i < total_nodes; ++i) {
265  in_buffer_util += in_buffers[i].size();
266  out_buffer_util += out_buffers[i].size();
267  }
268 
269  cycles++;
270 }

325 void LocalInterconnect::Push(unsigned input_deviceID, unsigned output_deviceID,
326  void* data, unsigned int size) {
327  unsigned subnet;
328  if (n_subnets == 1) {
329  subnet = 0;

332  subnet = 0;

334  subnet = 1;
335  }
336  }
337 
338  // it should have free buffer
339  // assume all the packets have size of one
340  // no flits are implemented
341  assert(net[subnet]->Has_Buffer_In(input_deviceID, 1));
342 
343  net[subnet]->Push(input_deviceID, output_deviceID, data, size);
344 }
