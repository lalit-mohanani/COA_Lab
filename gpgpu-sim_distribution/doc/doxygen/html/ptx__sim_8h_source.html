107  void mask_and(unsigned ms, unsigned ls) {
108  bits.ms &= ms;
109  bits.ls &= ls;
110  }
111 
112  void mask_or(unsigned ms, unsigned ls) {
113  bits.ms |= ms;
114  bits.ls |= ls;
115  }
116  int get_bit(unsigned bit) {
117  if (bit < 32)
118  return (bits.ls >> bit) & 1;
119  else
120  return (bits.ms >> (bit - 32)) & 1;
121  }

247  assert(m_valid);
248  m_sm_version_str = sm_ver;
249  check_target_extension(ext);

252  m_sm_version_valid = true;
253  }

255  assert(m_valid);

257  }

259  assert(m_valid && m_sm_version_valid);
260  return m_sm_version;
261  }
262  unsigned extensions() const {
263  assert(m_valid);
264  return m_ptx_extensions;
265  }
266 
267  private:
268  void check_target_extension(const char *ext) {
269  if (ext) {
270  if (!strcmp(ext, "texmode_independent"))
271  m_texmode_unified = false;
272  else if (!strcmp(ext, "texmode_unified"))
273  m_texmode_unified = true;
274  else if (!strcmp(ext, "map_f64_to_f32"))
275  m_map_f64_to_f32 = true;
276  else
277  abort();
278  }
279  }
