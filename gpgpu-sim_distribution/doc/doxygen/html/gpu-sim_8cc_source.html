742 kernel_info_t *gpgpu_sim::select_kernel() {
743  if (m_running_kernels[m_last_issued_kernel] &&
744  !m_running_kernels[m_last_issued_kernel]->no_more_ctas_to_run() &&
745  !m_running_kernels[m_last_issued_kernel]->m_kernel_TB_latency) {
746  unsigned launch_uid = m_running_kernels[m_last_issued_kernel]->get_uid();
747  if (std::find(m_executed_kernel_uids.begin(), m_executed_kernel_uids.end(),
748  launch_uid) == m_executed_kernel_uids.end()) {
749  m_running_kernels[m_last_issued_kernel]->start_cycle =
750  gpu_sim_cycle + gpu_tot_sim_cycle;
751  m_executed_kernel_uids.push_back(launch_uid);
752  m_executed_kernel_names.push_back(
753  m_running_kernels[m_last_issued_kernel]->name());
754  }
755  return m_running_kernels[m_last_issued_kernel];
756  }
757 
758  for (unsigned n = 0; n < m_running_kernels.size(); n++) {
759  unsigned idx =
760  (n + m_last_issued_kernel + 1) % m_config.max_concurrent_kernel;
761  if (kernel_more_cta_left(m_running_kernels[idx]) &&
762  !m_running_kernels[idx]->m_kernel_TB_latency) {
763  m_last_issued_kernel = idx;
764  m_running_kernels[idx]->start_cycle = gpu_sim_cycle + gpu_tot_sim_cycle;
765  // record this kernel for stat print if it is the first time this kernel
766  // is selected for execution
767  unsigned launch_uid = m_running_kernels[idx]->get_uid();
768  assert(std::find(m_executed_kernel_uids.begin(),
769  m_executed_kernel_uids.end(),
770  launch_uid) == m_executed_kernel_uids.end());
771  m_executed_kernel_uids.push_back(launch_uid);
772  m_executed_kernel_names.push_back(m_running_kernels[idx]->name());
773 
774  return m_running_kernels[idx];
775  }
776  }
777  return NULL;
778 }
