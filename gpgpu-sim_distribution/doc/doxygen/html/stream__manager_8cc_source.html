114 bool stream_operation::do_operation(gpgpu_sim *gpu) {
115  if (is_noop()) return true;
116 
117  assert(!m_done && m_stream);
118  if (g_debug_execution >= 3)
119  printf("GPGPU-Sim API: stream %u performing ", m_stream->get_uid());
120  switch (m_type) {
121  case stream_memcpy_host_to_device:
122  if (g_debug_execution >= 3) printf("memcpy host-to-device\n");
123  gpu->memcpy_to_gpu(m_device_address_dst, m_host_address_src, m_cnt);
124  m_stream->record_next_done();
125  break;
126  case stream_memcpy_device_to_host:
127  if (g_debug_execution >= 3) printf("memcpy device-to-host\n");
128  gpu->memcpy_from_gpu(m_host_address_dst, m_device_address_src, m_cnt);
129  m_stream->record_next_done();
130  break;
131  case stream_memcpy_device_to_device:
132  if (g_debug_execution >= 3) printf("memcpy device-to-device\n");
133  gpu->memcpy_gpu_to_gpu(m_device_address_dst, m_device_address_src, m_cnt);
134  m_stream->record_next_done();
135  break;
136  case stream_memcpy_to_symbol:
137  if (g_debug_execution >= 3) printf("memcpy to symbol\n");
138  gpu->gpgpu_ctx->func_sim->gpgpu_ptx_sim_memcpy_symbol(
139  m_symbol, m_host_address_src, m_cnt, m_offset, 1, gpu);
140  m_stream->record_next_done();
141  break;
142  case stream_memcpy_from_symbol:
143  if (g_debug_execution >= 3) printf("memcpy from symbol\n");
144  gpu->gpgpu_ctx->func_sim->gpgpu_ptx_sim_memcpy_symbol(
145  m_symbol, m_host_address_dst, m_cnt, m_offset, 0, gpu);
146  m_stream->record_next_done();
147  break;
148  case stream_kernel_launch:
149  if (m_sim_mode) { // Functional Sim
150  if (g_debug_execution >= 3) {
151  printf("kernel %d: \'%s\' transfer to GPU hardware scheduler\n",
152  m_kernel->get_uid(), m_kernel->name().c_str());
153  m_kernel->print_parent_info();
154  }
155  gpu->set_cache_config(m_kernel->name());
156  gpu->functional_launch(m_kernel);
157  } else { // Performance Sim
158  if (gpu->can_start_kernel() && m_kernel->m_launch_latency == 0) {
159  if (g_debug_execution >= 3) {
160  printf("kernel %d: \'%s\' transfer to GPU hardware scheduler\n",
161  m_kernel->get_uid(), m_kernel->name().c_str());
162  m_kernel->print_parent_info();
163  }
164  gpu->set_cache_config(m_kernel->name());
165  gpu->launch(m_kernel);
166  } else {
167  if (m_kernel->m_launch_latency) m_kernel->m_launch_latency--;
168  if (g_debug_execution >= 3)
169  printf(
170  "kernel %d: \'%s\', latency %u not ready to transfer to GPU "
171  "hardware scheduler\n",
172  m_kernel->get_uid(), m_kernel->name().c_str(),
173  m_kernel->m_launch_latency);
174  return false;
175  }
176  }
177  break;
178  case stream_event: {
179  printf("event update\n");
180  time_t wallclock = time((time_t *)NULL);
181  m_event->update(gpu->gpu_tot_sim_cycle, wallclock);
182  m_stream->record_next_done();
183  } break;
184  case stream_wait_event:
185  // only allows next op to go if event is done
186  // otherwise stays in the stream queue
187  printf("stream wait event processing...\n");
188  if (m_event->num_updates() >= m_cnt) {
189  printf("stream wait event done\n");
190  m_stream->record_next_done();
191  } else {
192  return false;
193  }
194  break;
195  default:
196  abort();
197  }
198  m_done = true;
199  fflush(stdout);
200  return true;
201 }

242  bool check = check_finished_kernel();
243  pthread_mutex_lock(&m_lock);
244  // if(check)m_gpu->print_stats();
245  stream_operation op = front();
246  if (!op.do_operation(m_gpu)) // not ready to execute
247  {
248  // cancel operation

250  unsigned grid_uid = op.get_kernel()->get_uid();
251  m_grid_id_to_stream.erase(grid_uid);
252  }
253  op.get_stream()->cancel_front();
254  }
255  pthread_mutex_unlock(&m_lock);
256  // pthread_mutex_lock(&m_lock);
257  // simulate a clock cycle on the GPU
258  return check;
259 }

318 stream_operation stream_manager::front() {
319  // called by gpu simulation thread
320  stream_operation result;
321  // if( concurrent_streams_empty() )
322  m_service_stream_zero = true;
323  if (m_service_stream_zero) {
324  if (!m_stream_zero.empty() && !m_stream_zero.busy()) {
325  result = m_stream_zero.next();
326  if (result.is_kernel()) {
327  unsigned grid_id = result.get_kernel()->get_uid();
328  m_grid_id_to_stream[grid_id] = &m_stream_zero;
329  }
330  } else {
331  m_service_stream_zero = false;
332  }
333  }
334  if (!m_service_stream_zero) {
335  std::list<struct CUstream_st *>::iterator s = m_last_stream;
336  if (m_last_stream == m_streams.end()) {
337  s = m_streams.begin();
338  } else {
339  s++;
340  }
341  for (size_t ii = 0; ii < m_streams.size(); ii++, s++) {
342  if (s == m_streams.end()) {
343  s = m_streams.begin();
344  }
345  m_last_stream = s;
346  CUstream_st *stream = *s;
347  if (!stream->busy() && !stream->empty()) {
348  result = stream->next();
349  if (result.is_kernel()) {
350  unsigned grid_id = result.get_kernel()->get_uid();
351  m_grid_id_to_stream[grid_id] = stream;
352  }
353  break;
354  }
355  }
356  }
357  return result;
358 }

431 void stream_manager::push(stream_operation op) {
432  struct CUstream_st *stream = op.get_stream();
433 
434  // block if stream 0 (or concurrency disabled) and pending concurrent
435  // operations exist
436  bool block = !stream || m_cuda_launch_blocking;
437  while (block) {
438  pthread_mutex_lock(&m_lock);
439  block = !concurrent_streams_empty();
440  pthread_mutex_unlock(&m_lock);
441  };
442 
443  pthread_mutex_lock(&m_lock);
444  if (!m_gpu->cycle_insn_cta_max_hit()) {
445  // Accept the stream operation if the maximum cycle/instruction/cta counts
446  // are not triggered
447  if (stream && !m_cuda_launch_blocking) {
448  stream->push(op);
449  } else {
450  op.set_stream(&m_stream_zero);
451  m_stream_zero.push(op);
452  }
453  } else {
454  // Otherwise, ignore operation and continue
455  printf(
456  "GPGPU-Sim API: Maximum cycle, instruction, or CTA count hit. "
457  "Skipping:");
458  op.print(stdout);
459  printf("\n");
460  }
461  if (g_debug_execution >= 3) print_impl(stdout);
462  pthread_mutex_unlock(&m_lock);
463  if (m_cuda_launch_blocking || stream == NULL) {
464  unsigned int wait_amount = 100;
465  unsigned int wait_cap = 100000; // 100ms
466  while (!empty()) {
467  // sleep to prevent CPU hog by empty spin
468  // sleep time increased exponentially ensure fast response when needed
469  usleep(wait_amount);
470  wait_amount *= 2;
471  if (wait_amount > wait_cap) wait_amount = wait_cap;
472  }
473  }
474 }
