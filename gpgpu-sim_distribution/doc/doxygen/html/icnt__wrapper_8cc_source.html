163 void icnt_wrapper_init() {
164  switch (g_network_mode) {
165  case INTERSIM:
166  // FIXME: delete the object: may add icnt_done wrapper
167  g_icnt_interface = InterconnectInterface::New(g_network_config_filename);
168  icnt_create = intersim2_create;
169  icnt_init = intersim2_init;
170  icnt_has_buffer = intersim2_has_buffer;
171  icnt_push = intersim2_push;
172  icnt_pop = intersim2_pop;
173  icnt_transfer = intersim2_transfer;
174  icnt_busy = intersim2_busy;
175  icnt_display_stats = intersim2_display_stats;
176  icnt_display_overall_stats = intersim2_display_overall_stats;
177  icnt_display_state = intersim2_display_state;
178  icnt_get_flit_size = intersim2_get_flit_size;
179  break;
180  case LOCAL_XBAR:
181  g_localicnt_interface = LocalInterconnect::New(g_inct_config);
182  icnt_create = LocalInterconnect_create;
183  icnt_init = LocalInterconnect_init;
184  icnt_has_buffer = LocalInterconnect_has_buffer;
185  icnt_push = LocalInterconnect_push;
186  icnt_pop = LocalInterconnect_pop;
187  icnt_transfer = LocalInterconnect_transfer;
188  icnt_busy = LocalInterconnect_busy;
189  icnt_display_stats = LocalInterconnect_display_stats;
190  icnt_display_overall_stats = LocalInterconnect_display_overall_stats;
191  icnt_display_state = LocalInterconnect_display_state;
192  icnt_get_flit_size = LocalInterconnect_get_flit_size;
193  break;
194  default:
195  assert(0);
196  break;
197  }
