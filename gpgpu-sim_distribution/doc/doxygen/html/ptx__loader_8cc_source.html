84 void gpgpu_context::print_ptx_file(const char *p, unsigned source_num,
85  const char *filename) {
86  printf("\nGPGPU-Sim PTX: file _%u.ptx contents:\n\n", source_num);
87  char *s = strdup(p);
88  char *t = s;
89  unsigned n = 1;
90  while (*t != '\0') {
91  char *u = t;
92  while ((*u != '\n') && (*u != '\0')) u++;
93  unsigned last = (*u == '\0');
94  *u = '\0';
95  const ptx_instruction *pI = ptx_parser->ptx_instruction_lookup(filename, n);
96  char pc[64];
97  if (pI && pI->get_PC())
98  snprintf(pc, 64, "%4u", pI->get_PC());
99  else
100  snprintf(pc, 64, " ");
101  printf(" _%u.ptx %4u (pc=%s): %s\n", source_num, n, pc, t);
102  if (last) break;
103  t = u + 1;
104  n++;
105  }
106  free(s);
107  fflush(stdout);
108 }

192  fprintf(ptxfile, "%s", p);
193  fclose(ptxfile);
194  abort();
195  exit(40);
196  }
197  ptx_lex_destroy(ptx_parser->scanner);
198 
199  if (g_debug_execution >= 100) print_ptx_file(p, source_num, buf);
200 
201  printf("GPGPU-Sim PTX: finished parsing EMBEDDED .ptx file %s\n", buf);

212 void fix_duplicate_errors(char fname2[1024]) {
213  char tempfile[1024] = "_temp_ptx";
214  char commandline[1024];
215 
216  // change the name of the ptx file to _temp_ptx
217  snprintf(commandline, 1024, "mv %s %s", fname2, tempfile);
218  printf("Running: %s\n", commandline);
219  int result = system(commandline);
220  if (result != 0) {
221  fprintf(stderr,
222  "GPGPU-Sim PTX: ERROR ** while changing filename from %s to %s",
223  fname2, tempfile);
224  exit(1);
225  }
226 
227  // store all of the ptx into a char array
228  FILE *ptxsource = fopen(tempfile, "r");
229  fseek(ptxsource, 0, SEEK_END);
230  long filesize = ftell(ptxsource);
231  rewind(ptxsource);
232  char *ptxdata = (char *)malloc((filesize + 1) * sizeof(char));
233  // Fail if we do not read the file
234  assert(fread(ptxdata, filesize, 1, ptxsource) == 1);
235  fclose(ptxsource);
236 
237  FILE *ptxdest = fopen(fname2, "w");
238  std::map<unsigned, const char *> duplicate = get_duplicate();
239  unsigned offset;
240  unsigned oldlinenum = 1;
241  unsigned linenum;
242  char *startptr = ptxdata;
243  char *funcptr;
244  char *tempptr = ptxdata - 1;
245  char *lineptr = ptxdata - 1;
246 
247  // recreate the ptx file without duplications
248  for (std::map<unsigned, const char *>::iterator iter = duplicate.begin();
249  iter != duplicate.end(); iter++) {
250  // find the line of the next error
251  linenum = iter->first;
252  for (int i = oldlinenum; i < linenum; i++) {
253  lineptr = strchr(lineptr + 1, '\n');
254  }
255 
256  // find the end of the current section to be copied over
257  // then find the start of the next section that will be copied
258  if (strcmp("function", iter->second) == 0) {
259  // get location of most recent .func
260  while (tempptr < lineptr && tempptr != NULL) {
261  funcptr = tempptr;
262  tempptr = strstr(funcptr + 1, ".func");
263  }
264 
265  // get the start of the previous line
266  offset = 0;
267  while (*(funcptr - offset) != '\n') offset++;
268 
269  fwrite(startptr, sizeof(char), funcptr - offset + 1 - startptr, ptxdest);
270 
271  // find next location of startptr
272  if (*(lineptr + 3) == ';') {
273  // for function definitions
274  startptr = lineptr + 5;
275  } else if (*(lineptr + 3) == '{') {
276  // for functions enclosed with curly brackets
277  offset = 5;
278  unsigned bracket = 1;
279  while (bracket != 0) {
280  if (*(lineptr + offset) == '{')
281  bracket++;
282  else if (*(lineptr + offset) == '}')
283  bracket--;
284  offset++;
285  }
286  startptr = lineptr + offset + 1;
287  } else {
288  printf("GPGPU-Sim PTX: ERROR ** Unrecognized function format\n");
289  abort();
290  }
291  } else if (strcmp("variable", iter->second) == 0) {
292  fwrite(startptr, sizeof(char), (int)(lineptr + 1 - startptr), ptxdest);
293 
294  // find next location of startptr
295  offset = 1;
296  while (*(lineptr + offset) != '\n') offset++;
297  startptr = lineptr + offset + 1;
298  } else {
299  printf("GPGPU-Sim PTX: ERROR ** Unsupported duplicate type: %s\n",
300  iter->second);
301  }
302 
303  oldlinenum = linenum;
304  }
305  // copy over the rest of the file
306  fwrite(startptr, sizeof(char), ptxdata + filesize - startptr, ptxdest);
307 
308  // cleanup
309  free(ptxdata);
310  fclose(ptxdest);
311  snprintf(commandline, 1024, "rm -f %s", tempfile);
312  printf("Running: %s\n", commandline);
313  result = system(commandline);
314  if (result != 0) {
315  fprintf(stderr, "GPGPU-Sim PTX: ERROR ** while deleting %s", tempfile);
316  exit(1);
317  }
318 }

321 char *get_app_binary_name() {
322  char exe_path[1025];
323  char *self_exe_path;
324 #ifdef __APPLE__
325  // AMRUTH: get apple device and check the result.

334

342  printf("self exe links to: %s\n", self_exe_path);
