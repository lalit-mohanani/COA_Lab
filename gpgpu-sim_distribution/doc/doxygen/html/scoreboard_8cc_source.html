58 void Scoreboard::reserveRegister(unsigned wid, unsigned regnum) {
59  if (!(reg_table[wid].find(regnum) == reg_table[wid].end())) {
60  printf(
61  "Error: trying to reserve an already reserved register (sid=%d, "
62  "wid=%d, regnum=%d).",
63  m_sid, wid, regnum);
64  abort();
65  }
66  SHADER_DPRINTF(SCOREBOARD, "Reserved Register - warp:%d, reg: %d\n", wid,
67  regnum);

83 void Scoreboard::reserveRegisters(const class warp_inst_t* inst) {
84  for (unsigned r = 0; r < MAX_OUTPUT_VALUES; r++) {
85  if (inst->out[r] > 0) {
86  reserveRegister(inst->warp_id(), inst->out[r]);
87  SHADER_DPRINTF(SCOREBOARD, "Reserved register - warp:%d, reg: %d\n",
88  inst->warp_id(), inst->out[r]);
89  }
90  }
91 
92  // Keep track of long operations
93  if (inst->is_load() && (inst->space.get_type() == global_space ||
94  inst->space.get_type() == local_space ||
95  inst->space.get_type() == param_space_kernel ||
96  inst->space.get_type() == param_space_local ||
97  inst->space.get_type() == param_space_unclassified ||
98  inst->space.get_type() == tex_space)) {
99  for (unsigned r = 0; r < MAX_OUTPUT_VALUES; r++) {
100  if (inst->out[r] > 0) {
101  SHADER_DPRINTF(SCOREBOARD, "New longopreg marked - warp:%d, reg: %d\n",
102  inst->warp_id(), inst->out[r]);
103  longopregs[inst->warp_id()].insert(inst->out[r]);
104  }
105  }
106  }
107 }

109 // Release registers for an instruction
110 void Scoreboard::releaseRegisters(const class warp_inst_t* inst) {
111  for (unsigned r = 0; r < MAX_OUTPUT_VALUES; r++) {
112  if (inst->out[r] > 0) {
113  SHADER_DPRINTF(SCOREBOARD, "Register Released - warp:%d, reg: %d\n",
114  inst->warp_id(), inst->out[r]);
115  releaseRegister(inst->warp_id(), inst->out[r]);
116  longopregs[inst->warp_id()].erase(inst->out[r]);
117  }
118  }

128 bool Scoreboard::checkCollision(unsigned wid, const class inst_t* inst) const {
129  // Get list of all input and output registers
130  std::set<int> inst_regs;
131 
132  for (unsigned iii = 0; iii < inst->outcount; iii++)
133  inst_regs.insert(inst->out[iii]);
134 
135  for (unsigned jjj = 0; jjj < inst->incount; jjj++)
136  inst_regs.insert(inst->in[jjj]);
137 
138  if (inst->pred > 0) inst_regs.insert(inst->pred);
139  if (inst->ar1 > 0) inst_regs.insert(inst->ar1);
140  if (inst->ar2 > 0) inst_regs.insert(inst->ar2);
141 
142  // Check for collision, get the intersection of reserved registers and
143  // instruction registers
144  std::set<int>::const_iterator it2;
145  for (it2 = inst_regs.begin(); it2 != inst_regs.end(); it2++)
146  if (reg_table[wid].find(*it2) != reg_table[wid].end()) {
147  return true;
148  }
149  return false;
150 }
