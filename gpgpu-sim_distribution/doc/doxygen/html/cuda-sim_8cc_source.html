158 unsigned int intLOGB2(unsigned int v) {
159  unsigned int shift;
160  unsigned int r;
161 
162  r = 0;
163 
164  shift = ((v & 0xFFFF0000) != 0) << 4;
165  v >>= shift;
166  r |= shift;
167  shift = ((v & 0xFF00) != 0) << 3;
168  v >>= shift;
169  r |= shift;
170  shift = ((v & 0xF0) != 0) << 2;
171  v >>= shift;
172  r |= shift;
173  shift = ((v & 0xC) != 0) << 1;
174  v >>= shift;
175  r |= shift;
176  shift = ((v & 0x2) != 0) << 0;
177  v >>= shift;
178  r |= shift;
179 
180  return r;
181 }

183 void gpgpu_t::gpgpu_ptx_sim_bindTextureToArray(
184  const struct textureReference *texref, const struct cudaArray *array) {
185  std::string texname = gpgpu_ptx_sim_findNamefromTexture(texref);
186 
187  std::map<std::string, const struct cudaArray *>::const_iterator t =
188  m_NameToCudaArray.find(texname);
189  // check that there's nothing there first
190  if (t != m_NameToCudaArray.end()) {
191  printf(
192  "GPGPU-Sim PTX: Warning: binding to texref associated with %s, which "
193  "was previously bound.\nImplicitly unbinding texref associated to %s "
194  "first\n",
195  texname.c_str(), texname.c_str());
196  }
197  m_NameToCudaArray[texname] = array;
198  unsigned int texel_size_bits =
199  array->desc.w + array->desc.x + array->desc.y + array->desc.z;
200  unsigned int texel_size = texel_size_bits / 8;
201  unsigned int Tx, Ty;
202  int r;
203 
204  printf("GPGPU-Sim PTX: texel size = %d\n", texel_size);
205  printf("GPGPU-Sim PTX: texture cache linesize = %d\n",
206  m_function_model_config.get_texcache_linesize());
207  // first determine base Tx size for given linesize
208  switch (m_function_model_config.get_texcache_linesize()) {
209  case 16:
210  Tx = 4;
211  break;
212  case 32:
213  Tx = 8;
214  break;
215  case 64:
216  Tx = 8;
217  break;
218  case 128:
219  Tx = 16;
220  break;
221  case 256:
222  Tx = 16;
223  break;
224  default:
225  printf(
226  "GPGPU-Sim PTX: Line size of %d bytes currently not supported.\n",
227  m_function_model_config.get_texcache_linesize());
228  assert(0);
229  break;
230  }
231  r = texel_size >> 2;
232  // modify base Tx size to take into account size of each texel in bytes
233  while (r != 0) {
234  Tx = Tx >> 1;
235  r = r >> 2;
236  }
237  // by now, got the correct Tx size, calculate correct Ty size
238  Ty = m_function_model_config.get_texcache_linesize() / (Tx * texel_size);
239 
240  printf(
241  "GPGPU-Sim PTX: Tx = %d; Ty = %d, Tx_numbits = %d, Ty_numbits = %d\n",
242  Tx, Ty, intLOGB2(Tx), intLOGB2(Ty));
243  printf("GPGPU-Sim PTX: Texel size = %d bytes; texel_size_numbits = %d\n",
244  texel_size, intLOGB2(texel_size));
245  printf(
246  "GPGPU-Sim PTX: Binding texture to array starting at devPtr32 = 0x%x\n",
247  array->devPtr32);
248  printf("GPGPU-Sim PTX: Texel size = %d bytes\n", texel_size);
249  struct textureInfo *texInfo =
250  (struct textureInfo *)malloc(sizeof(struct textureInfo));
251  texInfo->Tx = Tx;
252  texInfo->Ty = Ty;
253  texInfo->Tx_numbits = intLOGB2(Tx);
254  texInfo->Ty_numbits = intLOGB2(Ty);
255  texInfo->texel_size = texel_size;
256  texInfo->texel_size_numbits = intLOGB2(texel_size);
257  m_NameToTextureInfo[texname] = texInfo;
258 }

270 void function_info::ptx_assemble() {
271  if (m_assembled) {
272  return;
273  }
274 
275  // get the instructions into instruction memory...
276  unsigned num_inst = m_instructions.size();
277  m_instr_mem_size = MAX_INST_SIZE * (num_inst + 1);
278  m_instr_mem = new ptx_instruction *[m_instr_mem_size];
279 
280  printf("GPGPU-Sim PTX: instruction assembly for function \'%s\'... ",
281  m_name.c_str());
282  fflush(stdout);
283  std::list<ptx_instruction *>::iterator i;
284 
285  addr_t PC =
286  gpgpu_ctx->func_sim->g_assemble_code_next_pc; // globally unique address
287  // (across functions)
288  // start function on an aligned address
289  for (unsigned i = 0; i < (PC % MAX_INST_SIZE); i++)
290  gpgpu_ctx->s_g_pc_to_insn.push_back((ptx_instruction *)NULL);
291  PC += PC % MAX_INST_SIZE;
292  m_start_PC = PC;
293 
294  addr_t n = 0; // offset in m_instr_mem
295  // Why s_g_pc_to_insn.size() is needed to reserve additional memory for insts?
296  // reserve is cumulative. s_g_pc_to_insn.reserve(s_g_pc_to_insn.size() +
297  // MAX_INST_SIZE*m_instructions.size());
298  gpgpu_ctx->s_g_pc_to_insn.reserve(MAX_INST_SIZE * m_instructions.size());
299  for (i = m_instructions.begin(); i != m_instructions.end(); i++) {
300  ptx_instruction *pI = *i;
301  if (pI->is_label()) {
302  const symbol *l = pI->get_label();
303  labels[l->name()] = n;
304  } else {
305  gpgpu_ctx->func_sim->g_pc_to_finfo[PC] = this;
306  m_instr_mem[n] = pI;
307  gpgpu_ctx->s_g_pc_to_insn.push_back(pI);
308  assert(pI == gpgpu_ctx->s_g_pc_to_insn[PC]);
309  pI->set_m_instr_mem_index(n);
310  pI->set_PC(PC);
311  assert(pI->inst_size() <= MAX_INST_SIZE);
312  for (unsigned i = 1; i < pI->inst_size(); i++) {
313  gpgpu_ctx->s_g_pc_to_insn.push_back((ptx_instruction *)NULL);
314  m_instr_mem[n + i] = NULL;
315  }
316  n += pI->inst_size();
317  PC += pI->inst_size();
318  }
319  }
320  gpgpu_ctx->func_sim->g_assemble_code_next_pc = PC;
321  for (unsigned ii = 0; ii < n;
322  ii += m_instr_mem[ii]->inst_size()) { // handle branch instructions
323  ptx_instruction *pI = m_instr_mem[ii];
324  if (pI->get_opcode() == BRA_OP || pI->get_opcode() == BREAKADDR_OP ||
325  pI->get_opcode() == CALLP_OP) {
326  operand_info &target = pI->dst(); // get operand, e.g. target name
327  if (labels.find(target.name()) == labels.end()) {
328  printf(
329  "GPGPU-Sim PTX: Loader error (%s:%u): Branch label \"%s\" does not "
330  "appear in assembly code.",
331  pI->source_file(), pI->source_line(), target.name().c_str());
332  abort();
333  }
334  unsigned index = labels[target.name()]; // determine address from name
335  unsigned PC = m_instr_mem[index]->get_PC();
336  m_symtab->set_label_address(target.get_symbol(), PC);
337  target.set_type(label_t);
338  }
339  }
340  m_n = n;
341  printf(" done.\n");
342  fflush(stdout);
343 
344  // disable pdom analysis here and do it at runtime
345 #if 0
346  printf("GPGPU-Sim PTX: finding reconvergence points for \'%s\'...\n", m_name.c_str() );
347  create_basic_blocks();
348  connect_basic_blocks();
349  bool modified = false;
350  do {
351  find_dominators();
352  find_idominators();
353  modified = connect_break_targets();
354  } while (modified == true);
355 
356  if ( g_debug_execution>=50 ) {
357  print_basic_blocks();
358  print_basic_block_links();
359  print_basic_block_dot();
360  }
361  if ( g_debug_execution>=2 ) {
362  print_dominators();
363  }
364  find_postdominators();
365  find_ipostdominators();
366  if ( g_debug_execution>=50 ) {
367  print_postdominators();
368  print_ipostdominators();
369  }
370 
371  printf("GPGPU-Sim PTX: pre-decoding instructions for \'%s\'...\n", m_name.c_str() );
372  for ( unsigned ii=0; ii < n; ii += m_instr_mem[ii]->inst_size() ) { // handle branch instructions
373  ptx_instruction *pI = m_instr_mem[ii];
374  pI->pre_decode();
375  }
376  printf("GPGPU-Sim PTX: ... done pre-decoding instructions for \'%s\'.\n", m_name.c_str() );
377  fflush(stdout);
378 
379  m_assembled = true;
380 #endif
381 }

1000 void ptx_instruction::pre_decode() {
1001  pc = m_PC;
1002  isize = m_inst_size;
1003  for (unsigned i = 0; i < MAX_OUTPUT_VALUES; i++) {
1004  out[i] = 0;
1005  }
1006  for (unsigned i = 0; i < MAX_INPUT_VALUES; i++) {
1007  in[i] = 0;
1008  }
1009  incount = 0;
1010  outcount = 0;
1011  is_vectorin = 0;
1012  is_vectorout = 0;
1013  std::fill_n(arch_reg.src, MAX_REG_OPERANDS, -1);
1014  std::fill_n(arch_reg.dst, MAX_REG_OPERANDS, -1);
1015  pred = 0;
1016  ar1 = 0;
1017  ar2 = 0;
1018  space = m_space_spec;
1019  memory_op = no_memory_op;
1020  data_size = 0;
1021  if (has_memory_read() || has_memory_write()) {
1022  unsigned to_type = get_type();
1023  data_size = datatype2size(to_type);
1024  memory_op = has_memory_read() ? memory_load : memory_store;
1025  }
1026 
1027  bool has_dst = false;
1028 
1029  switch (get_opcode()) {
1030 #define OP_DEF(OP, FUNC, STR, DST, CLASSIFICATION) \
1031  case OP: \
1032  has_dst = (DST != 0); \
1033  break;
1034 #define OP_W_DEF(OP, FUNC, STR, DST, CLASSIFICATION) \
1035  case OP: \
1036  has_dst = (DST != 0); \
1037  break;
1038 #include "opcodes.def"
1039 #undef OP_DEF
1040 #undef OP_W_DEF
1041  default:
1042  printf("Execution error: Invalid opcode (0x%x)\n", get_opcode());
1043  break;
1044  }
1045 
1046  switch (m_cache_option) {
1047  case CA_OPTION:
1048  cache_op = CACHE_ALL;
1049  break;
1050  case NC_OPTION:
1051  cache_op = CACHE_L1;
1052  break;
1053  case CG_OPTION:
1054  cache_op = CACHE_GLOBAL;
1055  break;
1056  case CS_OPTION:
1057  cache_op = CACHE_STREAMING;
1058  break;
1059  case LU_OPTION:
1060  cache_op = CACHE_LAST_USE;
1061  break;
1062  case CV_OPTION:
1063  cache_op = CACHE_VOLATILE;
1064  break;
1065  case WB_OPTION:
1066  cache_op = CACHE_WRITE_BACK;
1067  break;
1068  case WT_OPTION:
1069  cache_op = CACHE_WRITE_THROUGH;
1070  break;
1071  default:
1072  // if( m_opcode == LD_OP || m_opcode == LDU_OP )
1073  if (m_opcode == MMA_LD_OP || m_opcode == LD_OP || m_opcode == LDU_OP)
1074  cache_op = CACHE_ALL;
1075  // else if( m_opcode == ST_OP )
1076  else if (m_opcode == MMA_ST_OP || m_opcode == ST_OP)
1077  cache_op = CACHE_WRITE_BACK;
1078  else if (m_opcode == ATOM_OP)
1079  cache_op = CACHE_GLOBAL;
1080  break;
1081  }
1082 
1083  set_opcode_and_latency();
1084  set_bar_type();
1085  // Get register operands
1086  int n = 0, m = 0;
1087  ptx_instruction::const_iterator opr = op_iter_begin();
1088  for (; opr != op_iter_end(); opr++, n++) { // process operands
1089  const operand_info &o = *opr;
1090  if (has_dst && n == 0) {
1091  // Do not set the null register "_" as an architectural register
1092  if (o.is_reg() && !o.is_non_arch_reg()) {
1093  out[0] = o.reg_num();
1094  arch_reg.dst[0] = o.arch_reg_num();
1095  } else if (o.is_vector()) {
1096  is_vectorin = 1;
1097  unsigned num_elem = o.get_vect_nelem();
1098  if (num_elem >= 1) out[0] = o.reg1_num();
1099  if (num_elem >= 2) out[1] = o.reg2_num();
1100  if (num_elem >= 3) out[2] = o.reg3_num();
1101  if (num_elem >= 4) out[3] = o.reg4_num();
1102  if (num_elem >= 5) out[4] = o.reg5_num();
1103  if (num_elem >= 6) out[5] = o.reg6_num();
1104  if (num_elem >= 7) out[6] = o.reg7_num();
1105  if (num_elem >= 8) out[7] = o.reg8_num();
1106  for (int i = 0; i < num_elem; i++) arch_reg.dst[i] = o.arch_reg_num(i);
1107  }
1108  } else {
1109  if (o.is_reg() && !o.is_non_arch_reg()) {
1110  int reg_num = o.reg_num();
1111  arch_reg.src[m] = o.arch_reg_num();
1112  switch (m) {
1113  case 0:
1114  in[0] = reg_num;
1115  break;
1116  case 1:
1117  in[1] = reg_num;
1118  break;
1119  case 2:
1120  in[2] = reg_num;
1121  break;
1122  default:
1123  break;
1124  }
1125  m++;
1126  } else if (o.is_vector()) {
1127  // assert(m == 0); //only support 1 vector operand (for textures) right
1128  // now
1129  is_vectorout = 1;
1130  unsigned num_elem = o.get_vect_nelem();
1131  if (num_elem >= 1) in[m + 0] = o.reg1_num();
1132  if (num_elem >= 2) in[m + 1] = o.reg2_num();
1133  if (num_elem >= 3) in[m + 2] = o.reg3_num();
1134  if (num_elem >= 4) in[m + 3] = o.reg4_num();
1135  if (num_elem >= 5) in[m + 4] = o.reg5_num();
1136  if (num_elem >= 6) in[m + 5] = o.reg6_num();
1137  if (num_elem >= 7) in[m + 6] = o.reg7_num();
1138  if (num_elem >= 8) in[m + 7] = o.reg8_num();
1139  for (int i = 0; i < num_elem; i++)
1140  arch_reg.src[m + i] = o.arch_reg_num(i);
1141  m += num_elem;
1142  }
1143  }
1144  }
1145 
1146  // Setting number of input and output operands which is required for
1147  // scoreboard check
1148  for (int i = 0; i < MAX_OUTPUT_VALUES; i++)
1149  if (out[i] > 0) outcount++;
1150 
1151  for (int i = 0; i < MAX_INPUT_VALUES; i++)
1152  if (in[i] > 0) incount++;
1153 
1154  // Get predicate
1155  if (has_pred()) {
1156  const operand_info &p = get_pred();
1157  pred = p.reg_num();
1158  }
1159 
1160  // Get address registers inside memory operands.
1161  // Assuming only one memory operand per instruction,
1162  // and maximum of two address registers for one memory operand.
1163  if (has_memory_read() || has_memory_write()) {
1164  ptx_instruction::const_iterator op = op_iter_begin();
1165  for (; op != op_iter_end(); op++, n++) { // process operands
1166  const operand_info &o = *op;
1167 
1168  if (o.is_memory_operand()) {
1169  // We do not support the null register as a memory operand
1170  assert(!o.is_non_arch_reg());

1987  "GPGPU-Sim PTX simulator: thread exiting ctaid=(%u,%u,%u) "

1991  }

1994  *thread_info = NULL;

2008  return 0; // finished!

2012  return 0;
2013  }
2014

2016  printf("GPGPU-Sim PTX simulator: STARTING THREAD ALLOCATION --> \n");

2018  }

2023  memory_space *sstarr_mem = NULL;
2024

2027  assert(max_cta_per_sm > 0);

2039  snprintf(buf, 512, "shared_%u", sid);
2040  shared_mem = new memory_space_impl<16 * 1024>(buf, 4);
2041  shared_memory_lookup[sm_idx] = shared_mem;
2042  snprintf(buf, 512, "sstarr_%u", sid);

2044  sstarr_memory_lookup[sm_idx] = sstarr_mem;

2046  ptx_cta_lookup[sm_idx] = cta_info;
2047  } else {

2049  printf(" <CTA realloc> : sm_idx=%u sid=%u max_cta_per_sm=%u\n", sm_idx,
2050  sid, max_cta_per_sm);
2051  }
2052  shared_mem = shared_memory_lookup[sm_idx];
2053  sstarr_mem = sstarr_memory_lookup[sm_idx];
2054  cta_info = ptx_cta_lookup[sm_idx];
2055  cta_info->check_cta_thread_status_and_reset();
2056  }
2057 
2058  std::map<unsigned, memory_space *> &local_mem_lookup =
2059  local_memory_lookup[sid];
2060  while (kernel.more_threads_in_cta()) {
2061  dim3 ctaid3d = kernel.get_next_cta_id();
2062  unsigned new_tid = kernel.get_next_thread_id();
2063  dim3 tid3d = kernel.get_next_thread_id_3d();

2065  new_tid += tid;

2067  ptx_warp_info *warp_info = NULL;
2068  if (ptx_warp_lookup.find(hw_warp_id) == ptx_warp_lookup.end()) {
2069  warp_info = new ptx_warp_info();
2070  ptx_warp_lookup[hw_warp_id] = warp_info;
2071  } else {
2072  warp_info = ptx_warp_lookup[hw_warp_id];
2073  }
2074  thd->m_warp_info = warp_info;
2075 
2076  memory_space *local_mem = NULL;
2077  std::map<unsigned, memory_space *>::iterator l =
2078  local_mem_lookup.find(new_tid);
2079  if (l != local_mem_lookup.end()) {
2080  local_mem = l->second;
2081  } else {
2082  char buf[512];
2083  snprintf(buf, 512, "local_%u_%u", sid, new_tid);
2084  local_mem = new memory_space_impl<32>(buf, 32);
2085  local_mem_lookup[new_tid] = local_mem;

2089  thd->set_ntid(kernel.get_cta_dim());

2102  cta_info->add_thread(thd);
2103  thd->m_local_mem = local_mem;

2105  printf(
2106  "GPGPU-Sim PTX simulator: allocating thread ctaid=(%u,%u,%u) "
2107  "tid=(%u,%u,%u) @ 0x%Lx\n",
2108  ctaid3d.x, ctaid3d.y, ctaid3d.z, tid3d.x, tid3d.y, tid3d.z,

2110  fflush(stdout);

2115  printf("GPGPU-Sim PTX simulator: <-- FINISHING THREAD ALLOCATION\n");

2120

2122  *thread_info = active_threads.front();
2123  (*thread_info)
2124  ->init(gpu, core, sid, hw_cta_id, hw_warp_id, tid,
2125  isInFunctionalSimulationMode);
2126  active_threads.pop_front();
2127  return 1;
2128 }
