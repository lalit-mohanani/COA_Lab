46 void *gpgpu_sim_thread_sequential(void *ctx_ptr) {
47  gpgpu_context *ctx = (gpgpu_context *)ctx_ptr;
48  // at most one kernel running at a time
49  bool done;
50  do {
51  sem_wait(&(ctx->the_gpgpusim->g_sim_signal_start));
52  done = true;
53  if (ctx->the_gpgpusim->g_the_gpu->get_more_cta_left()) {
54  done = false;
55  ctx->the_gpgpusim->g_the_gpu->init();
56  while (ctx->the_gpgpusim->g_the_gpu->active()) {
57  ctx->the_gpgpusim->g_the_gpu->cycle();
58  ctx->the_gpgpusim->g_the_gpu->deadlock_check();
59  }
60  ctx->the_gpgpusim->g_the_gpu->print_stats();
61  ctx->the_gpgpusim->g_the_gpu->update_stats();
62  ctx->print_simulation_time();
63  }
64  sem_post(&(ctx->the_gpgpusim->g_sim_signal_finish));
65  } while (!done);
66  sem_post(&(ctx->the_gpgpusim->g_sim_signal_exit));
67  return NULL;
68 }

75 void *gpgpu_sim_thread_concurrent(void *ctx_ptr) {
76  gpgpu_context *ctx = (gpgpu_context *)ctx_ptr;
77  atexit(termination_callback);
78  // concurrent kernel execution simulation thread
79  do {
80  if (g_debug_execution >= 3) {
81  printf(
82  "GPGPU-Sim: *** simulation thread starting and spinning waiting for "
83  "work ***\n");
84  fflush(stdout);
85  }
86  while (ctx->the_gpgpusim->g_stream_manager->empty_protected() &&
87  !ctx->the_gpgpusim->g_sim_done)
88  ;
89  if (g_debug_execution >= 3) {
90  printf("GPGPU-Sim: ** START simulation thread (detected work) **\n");
91  ctx->the_gpgpusim->g_stream_manager->print(stdout);
92  fflush(stdout);
93  }
94  pthread_mutex_lock(&(ctx->the_gpgpusim->g_sim_lock));
95  ctx->the_gpgpusim->g_sim_active = true;
96  pthread_mutex_unlock(&(ctx->the_gpgpusim->g_sim_lock));
97  bool active = false;
98  bool sim_cycles = false;
99  ctx->the_gpgpusim->g_the_gpu->init();
100  do {
101  // check if a kernel has completed
102  // launch operation on device if one is pending and can be run
103 
104  // Need to break this loop when a kernel completes. This was a
105  // source of non-deterministic behaviour in GPGPU-Sim (bug 147).
106  // If another stream operation is available, g_the_gpu remains active,
107  // causing this loop to not break. If the next operation happens to be
108  // another kernel, the gpu is not re-initialized and the inter-kernel
109  // behaviour may be incorrect. Check that a kernel has finished and
110  // no other kernel is currently running.
111  if (ctx->the_gpgpusim->g_stream_manager->operation(&sim_cycles) &&
112  !ctx->the_gpgpusim->g_the_gpu->active())
113  break;
114 
115  // functional simulation
116  if (ctx->the_gpgpusim->g_the_gpu->is_functional_sim()) {
117  kernel_info_t *kernel =
118  ctx->the_gpgpusim->g_the_gpu->get_functional_kernel();
119  assert(kernel);
120  ctx->the_gpgpusim->gpgpu_ctx->func_sim->gpgpu_cuda_ptx_sim_main_func(
121  *kernel);
122  ctx->the_gpgpusim->g_the_gpu->finish_functional_sim(kernel);
123  }
124 
125  // performance simulation
126  if (ctx->the_gpgpusim->g_the_gpu->active()) {
127  ctx->the_gpgpusim->g_the_gpu->cycle();
128  sim_cycles = true;
129  ctx->the_gpgpusim->g_the_gpu->deadlock_check();
130  } else {
131  if (ctx->the_gpgpusim->g_the_gpu->cycle_insn_cta_max_hit()) {
132  ctx->the_gpgpusim->g_stream_manager->stop_all_running_kernels();
133  ctx->the_gpgpusim->g_sim_done = true;
134  ctx->the_gpgpusim->break_limit = true;
135  }
136  }
137 
138  active = ctx->the_gpgpusim->g_the_gpu->active() ||
139  !(ctx->the_gpgpusim->g_stream_manager->empty_protected());
140 
141  } while (active && !ctx->the_gpgpusim->g_sim_done);
142  if (g_debug_execution >= 3) {
143  printf("GPGPU-Sim: ** STOP simulation thread (no work) **\n");
144  fflush(stdout);
145  }
146  if (sim_cycles) {
147  ctx->the_gpgpusim->g_the_gpu->print_stats();
148  ctx->the_gpgpusim->g_the_gpu->update_stats();
149  ctx->print_simulation_time();
150  }
151  pthread_mutex_lock(&(ctx->the_gpgpusim->g_sim_lock));
152  ctx->the_gpgpusim->g_sim_active = false;
153  pthread_mutex_unlock(&(ctx->the_gpgpusim->g_sim_lock));
154  } while (!ctx->the_gpgpusim->g_sim_done);
155 
156  printf("GPGPU-Sim: *** simulation thread exiting ***\n");
157  fflush(stdout);
158 
159  if (ctx->the_gpgpusim->break_limit) {
160  printf(
161  "GPGPU-Sim: ** break due to reaching the maximum cycles (or "
162  "instructions) **\n");
163  exit(1);
164  }
165 
166  sem_post(&(ctx->the_gpgpusim->g_sim_signal_exit));
167  return NULL;
168 }

248 void gpgpu_context::print_simulation_time() {
249  time_t current_time, difference, d, h, m, s;
250  current_time = time((time_t *)NULL);
251  difference = MAX(current_time - the_gpgpusim->g_simulation_starttime, 1);
252 
253  d = difference / (3600 * 24);
254  h = difference / 3600 - 24 * d;
255  m = difference / 60 - 60 * (h + 24 * d);
256  s = difference - 60 * (m + 60 * (h + 24 * d));
257 
258  fflush(stderr);
259  printf(
260  "\n\ngpgpu_simulation_time = %u days, %u hrs, %u min, %u sec (%u sec)\n",
261  (unsigned)d, (unsigned)h, (unsigned)m, (unsigned)s, (unsigned)difference);
262  printf("gpgpu_simulation_rate = %u (inst/sec)\n",
263  (unsigned)(the_gpgpusim->g_the_gpu->gpu_tot_sim_insn / difference));
264  const unsigned cycles_per_sec =
265  (unsigned)(the_gpgpusim->g_the_gpu->gpu_tot_sim_cycle / difference);
266  printf("gpgpu_simulation_rate = %u (cycle/sec)\n", cycles_per_sec);
267  printf("gpgpu_silicon_slowdown = %ux\n",
268  the_gpgpusim->g_the_gpu->shader_clock() * 1000 / cycles_per_sec);
269  fflush(stdout);
270 }
