143  void init() {
144  assert(gpgpu_dram_timing_opt);
145  if (strchr(gpgpu_dram_timing_opt, '=') == NULL) {
146  // dram timing option in ordered variables (legacy)
147  // Disabling bank groups if their values are not specified
148  nbkgrp = 1;
149  tCCDL = 0;
150  tRTPL = 0;
151  sscanf(gpgpu_dram_timing_opt, "%d:%d:%d:%d:%d:%d:%d:%d:%d:%d:%d:%d:%d:%d",
152  &nbk, &tCCD, &tRRD, &tRCD, &tRAS, &tRP, &tRC, &CL, &WL, &tCDLR,
153  &tWR, &nbkgrp, &tCCDL, &tRTPL);
154  } else {
155  // named dram timing options (unordered)
156  option_parser_t dram_opp = option_parser_create();
157 
158  option_parser_register(dram_opp, "nbk", OPT_UINT32, &nbk,
159  "number of banks", "");
160  option_parser_register(dram_opp, "CCD", OPT_UINT32, &tCCD,
161  "column to column delay", "");
162  option_parser_register(
163  dram_opp, "RRD", OPT_UINT32, &tRRD,
164  "minimal delay between activation of rows in different banks", "");
165  option_parser_register(dram_opp, "RCD", OPT_UINT32, &tRCD,
166  "row to column delay", "");
167  option_parser_register(dram_opp, "RAS", OPT_UINT32, &tRAS,
168  "time needed to activate row", "");
169  option_parser_register(dram_opp, "RP", OPT_UINT32, &tRP,
170  "time needed to precharge (deactivate) row", "");
171  option_parser_register(dram_opp, "RC", OPT_UINT32, &tRC, "row cycle time",
172  "");
173  option_parser_register(dram_opp, "CDLR", OPT_UINT32, &tCDLR,
174  "switching from write to read (changes tWTR)", "");
175  option_parser_register(dram_opp, "WR", OPT_UINT32, &tWR,
176  "last data-in to row precharge", "");
177 
178  option_parser_register(dram_opp, "CL", OPT_UINT32, &CL, "CAS latency",
179  "");
180  option_parser_register(dram_opp, "WL", OPT_UINT32, &WL, "Write latency",
181  "");
182 
183  // Disabling bank groups if their values are not specified
184  option_parser_register(dram_opp, "nbkgrp", OPT_UINT32, &nbkgrp,
185  "number of bank groups", "1");
186  option_parser_register(
187  dram_opp, "CCDL", OPT_UINT32, &tCCDL,
188  "column to column delay between accesses to different bank groups",
189  "0");
190  option_parser_register(
191  dram_opp, "RTPL", OPT_UINT32, &tRTPL,
192  "read to precharge delay between accesses to different bank groups",
193  "0");
194 
195  option_parser_delimited_string(dram_opp, gpgpu_dram_timing_opt, "=:;");
196  fprintf(stdout, "DRAM Timing Options:\n");
197  option_parser_print(dram_opp, stdout);
198  option_parser_destroy(dram_opp);
199  }
200 
201  int nbkt = nbk / nbkgrp;
202  unsigned i;
203  for (i = 0; nbkt > 0; i++) {
204  nbkt = nbkt >> 1;
205  }
206  bk_tag_length = i - 1;
207  assert(nbkgrp > 0 && "Number of bank groups cannot be zero");
208  tRCDWR = tRCD - (WL + 1);
209  if (elimnate_rw_turnaround) {
210  tRTW = 0;
211  tWTR = 0;
212  } else {
213  tRTW = (CL + (BL / data_command_freq_ratio) + 2 - WL);
214  tWTR = (WL + (BL / data_command_freq_ratio) + tCDLR);
215  }
216  tWTP = (WL + (BL / data_command_freq_ratio) + tWR);
217  dram_atom_size =
218  BL * busW * gpu_n_mem_per_ctrlr; // burst length x bus width x # chips
219  // per partition
220 
221  assert(m_n_sub_partition_per_memory_channel > 0);
222  assert((nbk % m_n_sub_partition_per_memory_channel == 0) &&
223  "Number of DRAM banks must be a perfect multiple of memory sub "
224  "partition");
225  m_n_mem_sub_partition = m_n_mem * m_n_sub_partition_per_memory_channel;
226  fprintf(stdout, "Total number of memory sub partition = %u\n",
227  m_n_mem_sub_partition);
228 
229  m_address_mapping.init(m_n_mem, m_n_sub_partition_per_memory_channel);
230  m_L2_config.init(&m_address_mapping);
231 
232  m_valid = true;
233 
234  sscanf(write_queue_size_opt, "%d:%d:%d",
235  &gpgpu_frfcfs_dram_write_queue_size, &write_high_watermark,
236  &write_low_watermark);
237  }

430 struct occupancy_stats {
431  occupancy_stats()
432  : aggregate_warp_slot_filled(0), aggregate_theoretical_warp_slots(0) {}
433  occupancy_stats(unsigned long long wsf, unsigned long long tws)
434  : aggregate_warp_slot_filled(wsf),
435  aggregate_theoretical_warp_slots(tws) {}
436 
437  unsigned long long aggregate_warp_slot_filled;
438  unsigned long long aggregate_theoretical_warp_slots;
439 
440  float get_occ_fraction() const {
441  return float(aggregate_warp_slot_filled) /
442  float(aggregate_theoretical_warp_slots);
443  }
444 
445  occupancy_stats &operator+=(const occupancy_stats &rhs) {
446  aggregate_warp_slot_filled += rhs.aggregate_warp_slot_filled;
447  aggregate_theoretical_warp_slots += rhs.aggregate_theoretical_warp_slots;
448  return *this;
449  }
450 
451  occupancy_stats operator+(const occupancy_stats &rhs) const {
452  return occupancy_stats(
453  aggregate_warp_slot_filled + rhs.aggregate_warp_slot_filled,
454  aggregate_theoretical_warp_slots +
455  rhs.aggregate_theoretical_warp_slots);
456  }
