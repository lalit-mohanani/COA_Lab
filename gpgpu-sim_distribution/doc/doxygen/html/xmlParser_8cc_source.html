939 XMLSTR fromXMLString(XMLCSTR s, int lo, XML *pXML) {
940  // This function is the opposite of the function "toXMLString". It decodes the
941  // escape sequences &amp;, &quot;, &apos;, &lt;, &gt; and replace them by the
942  // characters
943  // &,",',<,>. This function is used internally by the XML Parser. All the
944  // calls to the XML library will always gives you back "decoded" strings.
945  //
946  // in: string (s) and length (lo) of string
947  // out: new allocated string converted from xml
948  if (!s) return NULL;

950  int ll = 0, j;
951  XMLSTR d;
952  XMLCSTR ss = s;
953  XMLCharacterEntity *entity;
954  while ((lo > 0) && (*s)) {
955  if (*s == _CXML('&')) {
956  if ((lo > 2) && (s[1] == _CXML('#'))) {
957  s += 2;
958  lo -= 2;
959  if ((*s == _CXML('X')) || (*s == _CXML('x'))) {
960  s++;
961  lo--;
962  }
963  while ((*s) && (*s != _CXML(';')) && ((lo--) > 0)) s++;
964  if (*s != _CXML(';')) {
965  pXML->error = eXMLErrorUnknownCharacterEntity;
966  return NULL;
967  }
968  s++;
969  lo--;
970  } else {
971  entity = XMLEntities;
972  do {
973  if ((lo >= entity->l) && (xstrnicmp(s, entity->s, entity->l) == 0)) {
974  s += entity->l;
975  lo -= entity->l;
976  break;
977  }
978  entity++;
979  } while (entity->s);
980  if (!entity->s) {
981  pXML->error = eXMLErrorUnknownCharacterEntity;
982  return NULL;
983  }
984  }
985  } else {
986 #ifdef _XMLWIDECHAR
987  s++;
988  lo--;
989 #else

1593  pXML->nFirst = FALSE;
1594  status = eOutsideTag;

1619  case eTokenEquals:

1782  case eInsideTag:

1785  // are looking for.
