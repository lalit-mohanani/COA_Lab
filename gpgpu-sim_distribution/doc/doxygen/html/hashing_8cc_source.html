9 unsigned ipoly_hash_function(new_addr_type higher_bits, unsigned index,
10  unsigned bank_set_num) {
11  /*
12  * Set Indexing function from "Pseudo-randomly interleaved memory."
13  * Rau, B. R et al.
14  * ISCA 1991
15  * http://citeseerx.ist.psu.edu/viewdoc/download;jsessionid=348DEA37A3E440473B3C075EAABC63B6?doi=10.1.1.12.7149&rep=rep1&type=pdf
16  *
17  * equations are corresponding to IPOLY(37) and are adopted from:
18  * "Sacat: streaming-aware conflict-avoiding thrashing-resistant gpgpu
19  * cache management scheme." Khairy et al. IEEE TPDS 2017.
20  *
21  * equations for 16 banks are corresponding to IPOLY(5)
22  * equations for 32 banks are corresponding to IPOLY(37)
23  * equations for 64 banks are corresponding to IPOLY(67)
24  * To see all the IPOLY equations for all the degrees, see
25  * http://wireless-systems.ece.gatech.edu/6604/handouts/Peterson's%20Table.pdf
26  *
27  * We generate these equations using GF(2) arithmetic:
28  * http://www.ee.unb.ca/cgi-bin/tervo/calc.pl?num=&den=&f=d&e=1&m=1
29  *
30  * We go through all the strides 128 (10000000), 256 (100000000),... and
31  * do modular arithmetic in GF(2) Then, we create the H-matrix and group
32  * each bit together, for more info read the ISCA 1991 paper
33  *
34  * IPOLY hashing guarantees conflict-free for all 2^n strides which widely
35  * exit in GPGPU applications and also show good performance for other
36  * strides.
37  */
38  if (bank_set_num == 16) {
39  std::bitset<64> a(higher_bits);
40  std::bitset<4> b(index);
41  std::bitset<4> new_index(index);
42 
43  new_index[0] =
44  a[11] ^ a[10] ^ a[9] ^ a[8] ^ a[6] ^ a[4] ^ a[3] ^ a[0] ^ b[0];
45  new_index[1] =
46  a[12] ^ a[8] ^ a[7] ^ a[6] ^ a[5] ^ a[3] ^ a[1] ^ a[0] ^ b[1];
47  new_index[2] = a[9] ^ a[8] ^ a[7] ^ a[6] ^ a[4] ^ a[2] ^ a[1] ^ b[2];
48  new_index[3] = a[10] ^ a[9] ^ a[8] ^ a[7] ^ a[5] ^ a[3] ^ a[2] ^ b[3];
49 
50  return new_index.to_ulong();
51 
52  } else if (bank_set_num == 32) {
53  std::bitset<64> a(higher_bits);
54  std::bitset<5> b(index);
55  std::bitset<5> new_index(index);
56 
57  new_index[0] =
58  a[13] ^ a[12] ^ a[11] ^ a[10] ^ a[9] ^ a[6] ^ a[5] ^ a[3] ^ a[0] ^ b[0];
59  new_index[1] = a[14] ^ a[13] ^ a[12] ^ a[11] ^ a[10] ^ a[7] ^ a[6] ^ a[4] ^
60  a[1] ^ b[1];
61  new_index[2] =
62  a[14] ^ a[10] ^ a[9] ^ a[8] ^ a[7] ^ a[6] ^ a[3] ^ a[2] ^ a[0] ^ b[2];
63  new_index[3] =
64  a[11] ^ a[10] ^ a[9] ^ a[8] ^ a[7] ^ a[4] ^ a[3] ^ a[1] ^ b[3];
65  new_index[4] =
66  a[12] ^ a[11] ^ a[10] ^ a[9] ^ a[8] ^ a[5] ^ a[4] ^ a[2] ^ b[4];
67  return new_index.to_ulong();
68 
69  } else if (bank_set_num == 64) {
70  std::bitset<64> a(higher_bits);
71  std::bitset<6> b(index);
72  std::bitset<6> new_index(index);
73 
74  new_index[0] = a[18] ^ a[17] ^ a[16] ^ a[15] ^ a[12] ^ a[10] ^ a[6] ^ a[5] ^
75  a[0] ^ b[0];
76  new_index[1] = a[15] ^ a[13] ^ a[12] ^ a[11] ^ a[10] ^ a[7] ^ a[5] ^ a[1] ^
77  a[0] ^ b[1];
78  new_index[2] = a[16] ^ a[14] ^ a[13] ^ a[12] ^ a[11] ^ a[8] ^ a[6] ^ a[2] ^
79  a[1] ^ b[2];
80  new_index[3] = a[17] ^ a[15] ^ a[14] ^ a[13] ^ a[12] ^ a[9] ^ a[7] ^ a[3] ^
81  a[2] ^ b[3];
82  new_index[4] = a[18] ^ a[16] ^ a[15] ^ a[14] ^ a[13] ^ a[10] ^ a[8] ^ a[4] ^
83  a[3] ^ b[4];
84  new_index[5] =
85  a[17] ^ a[16] ^ a[15] ^ a[14] ^ a[11] ^ a[9] ^ a[5] ^ a[4] ^ b[5];
86  return new_index.to_ulong();
87  } else { /* Else incorrect number of channels for the hashing function */
88  assert(

90  "channels should be "

92  "numbers are not supported. Generate it by yourself! \n" &&
93  0);
94 
95  return 0;
