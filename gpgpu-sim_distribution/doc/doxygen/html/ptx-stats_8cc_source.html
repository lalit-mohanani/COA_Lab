49 // defining a PTX source line = filename + line number
50 class ptx_file_line {
51  public:
52  ptx_file_line(const char *s, int l) {
53  if (s == NULL)
54  st = "NULL_NAME";
55  else
56  st = s;
57  line = l;
58  }
59 
60  bool operator<(const ptx_file_line &other) const {
61  if (st == other.st) {
62  if (line < other.line)
63  return true;
64  else
65  return false;
66  } else {
67  return st < other.st;
68  }
69  }
70 
71  bool operator==(const ptx_file_line &other) const {
72  return (line == other.line) && (st == other.st);
73  }
74 
75  std::string st;
76  unsigned line;
77 };

126 // output statistics to a file
127 void ptx_stats::ptx_file_line_stats_write_file() {
128  // check if stat collection is turned on
129  if (enable_ptx_file_line_stats == 0) return;
130 
131  ptx_file_line_stats_map_t::iterator it;
132  FILE *pfile;
133 
134  pfile = fopen(ptx_line_stats_filename, "w");
135  fprintf(
136  pfile,
137  "kernel line : count latency dram_traffic smem_bk_conflicts smem_warp "
138  "gmem_access_generated gmem_warp exposed_latency warp_divergence\n");
139  for (it = ptx_file_line_stats_tracker.begin();
140  it != ptx_file_line_stats_tracker.end(); it++) {
141  fprintf(pfile, "%s %i : ", it->first.st.c_str(), it->first.line);
142  fprintf(pfile, "%lu ", it->second.exec_count);
143  fprintf(pfile, "%llu ", it->second.latency);
144  fprintf(pfile, "%llu ", it->second.dram_traffic);
145  fprintf(pfile, "%llu ", it->second.smem_n_way_bank_conflict_total);
146  fprintf(pfile, "%lu ", it->second.smem_warp_count);
147  fprintf(pfile, "%llu ", it->second.gmem_n_access_total);
148  fprintf(pfile, "%lu ", it->second.gmem_warp_count);
149  fprintf(pfile, "%llu ", it->second.exposed_latency);
150  fprintf(pfile, "%llu ", it->second.warp_divergence);
151  fprintf(pfile, "\n");
152  }
153  fflush(pfile);
154  fclose(pfile);
155 }

219 // a class that tracks the inflight memory instructions of a shader core
220 // and attributes exposed latency to those instructions when signaled to do so
221 class ptx_inflight_memory_insn_tracker {
222  public:
223  typedef std::map<const ptx_instruction *, int> insn_count_map;
224 
225  void add_count(const ptx_instruction *pInsn, int count = 1) {
226  ptx_inflight_memory_insns[pInsn] += count;
227  }
228 
229  void sub_count(const ptx_instruction *pInsn, int count = 1) {
230  insn_count_map::iterator i_insncount;
231  i_insncount = ptx_inflight_memory_insns.find(pInsn);
232 
233  assert(i_insncount != ptx_inflight_memory_insns.end());
234 
235  i_insncount->second -= count;
236 
237  if (i_insncount->second <= 0) {
238  ptx_inflight_memory_insns.erase(i_insncount);
239  }
240  }
241 
242  void attribute_exposed_latency(int count = 1) {
243  insn_count_map &exlat_insnmap = ptx_inflight_memory_insns;
244  insn_count_map::const_iterator i_exlatinsn;
245 
246  i_exlatinsn = exlat_insnmap.begin();
247  for (; i_exlatinsn != exlat_insnmap.end(); ++i_exlatinsn) {
248  const ptx_instruction *pInsn = i_exlatinsn->first;
249  ptx_file_line_stats &line_stats =
250  ptx_file_line_stats_tracker[ptx_file_line(pInsn->source_file(),
251  pInsn->source_line())];
252  line_stats.exposed_latency += count;
253  }
254  }
255 
256  insn_count_map ptx_inflight_memory_insns;
257 };
