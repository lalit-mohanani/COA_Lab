57 void power_mem_stat_t::init() {
58  shmem_read_access[CURRENT_STAT_IDX] =
59  m_core_stats->gpgpu_n_shmem_bank_access; // Shared memory access
60  shmem_read_access[PREV_STAT_IDX] =
61  (unsigned *)calloc(m_core_config->num_shader(), sizeof(unsigned));
62 
63  for (unsigned i = 0; i < NUM_STAT_IDX; ++i) {
64  core_cache_stats[i].clear();
65  l2_cache_stats[i].clear();
66 
67  n_cmd[i] = (unsigned *)calloc(m_config->m_n_mem, sizeof(unsigned));
68  n_activity[i] = (unsigned *)calloc(m_config->m_n_mem, sizeof(unsigned));
69  n_nop[i] = (unsigned *)calloc(m_config->m_n_mem, sizeof(unsigned));
70  n_act[i] = (unsigned *)calloc(m_config->m_n_mem, sizeof(unsigned));
71  n_pre[i] = (unsigned *)calloc(m_config->m_n_mem, sizeof(unsigned));
72  n_rd[i] = (unsigned *)calloc(m_config->m_n_mem, sizeof(unsigned));
73  n_wr[i] = (unsigned *)calloc(m_config->m_n_mem, sizeof(unsigned));
74  n_req[i] = (unsigned *)calloc(m_config->m_n_mem, sizeof(unsigned));
75 
76  // Interconnect stats
77  n_mem_to_simt[i] = (long *)calloc(m_core_config->n_simt_clusters,
78  sizeof(long)); // Counted at SM
79  n_simt_to_mem[i] = (long *)calloc(m_core_config->n_simt_clusters,
80  sizeof(long)); // Counted at SM
81  }

84 void power_mem_stat_t::save_stats() {
85  core_cache_stats[PREV_STAT_IDX] = core_cache_stats[CURRENT_STAT_IDX];
86  l2_cache_stats[PREV_STAT_IDX] = l2_cache_stats[CURRENT_STAT_IDX];
87 
88  for (unsigned i = 0; i < m_core_config->num_shader(); ++i) {
89  shmem_read_access[PREV_STAT_IDX][i] =
90  shmem_read_access[CURRENT_STAT_IDX][i]; // Shared memory access
91  }
92 
93  for (unsigned i = 0; i < m_config->m_n_mem; ++i) {
94  n_cmd[PREV_STAT_IDX][i] = n_cmd[CURRENT_STAT_IDX][i];
95  n_activity[PREV_STAT_IDX][i] = n_activity[CURRENT_STAT_IDX][i];
96  n_nop[PREV_STAT_IDX][i] = n_nop[CURRENT_STAT_IDX][i];
97  n_act[PREV_STAT_IDX][i] = n_act[CURRENT_STAT_IDX][i];
98  n_pre[PREV_STAT_IDX][i] = n_pre[CURRENT_STAT_IDX][i];
99  n_rd[PREV_STAT_IDX][i] = n_rd[CURRENT_STAT_IDX][i];
100  n_wr[PREV_STAT_IDX][i] = n_wr[CURRENT_STAT_IDX][i];
101  n_req[PREV_STAT_IDX][i] = n_req[CURRENT_STAT_IDX][i];
102  }
103 
104  for (unsigned i = 0; i < m_core_config->n_simt_clusters; i++) {
105  n_simt_to_mem[PREV_STAT_IDX][i] =
106  n_simt_to_mem[CURRENT_STAT_IDX][i]; // Interconnect
107  n_mem_to_simt[PREV_STAT_IDX][i] =
108  n_mem_to_simt[CURRENT_STAT_IDX][i]; // Interconnect
109  }

114 void power_mem_stat_t::print(FILE *fout) const {
115  fprintf(fout, "\n\n==========Power Metrics -- Memory==========\n");
116  unsigned total_mem_reads = 0;
117  unsigned total_mem_writes = 0;
118  for (unsigned i = 0; i < m_config->m_n_mem; ++i) {
119  total_mem_reads += n_rd[CURRENT_STAT_IDX][i];
120  total_mem_writes += n_wr[CURRENT_STAT_IDX][i];
121  }
122  fprintf(fout, "Total memory controller accesses: %u\n",
123  total_mem_reads + total_mem_writes);
124  fprintf(fout, "Total memory controller reads: %u\n", total_mem_reads);
125  fprintf(fout, "Total memory controller writes: %u\n", total_mem_writes);
126 
127  fprintf(fout, "Core cache stats:\n");
128  core_cache_stats->print_stats(fout);
129  fprintf(fout, "L2 cache stats:\n");
130  l2_cache_stats->print_stats(fout);
