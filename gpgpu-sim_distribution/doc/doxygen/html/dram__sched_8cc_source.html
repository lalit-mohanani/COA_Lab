35 frfcfs_scheduler::frfcfs_scheduler(const memory_config *config, dram_t *dm,
36  memory_stats_t *stats) {
37  m_config = config;
38  m_stats = stats;
39  m_num_pending = 0;
40  m_num_write_pending = 0;
41  m_dram = dm;
42  m_queue = new std::list<dram_req_t *>[m_config->nbk];
43  m_bins = new std::map<
44  unsigned, std::list<std::list<dram_req_t *>::iterator> >[m_config->nbk];
45  m_last_row =
46  new std::list<std::list<dram_req_t *>::iterator> *[m_config->nbk];
47  curr_row_service_time = new unsigned[m_config->nbk];
48  row_service_timestamp = new unsigned[m_config->nbk];
49  for (unsigned i = 0; i < m_config->nbk; i++) {
50  m_queue[i].clear();
51  m_bins[i].clear();
52  m_last_row[i] = NULL;
53  curr_row_service_time[i] = 0;
54  row_service_timestamp[i] = 0;
55  }
56  if (m_config->seperate_write_queue_enabled) {
57  m_write_queue = new std::list<dram_req_t *>[m_config->nbk];
58  m_write_bins = new std::map<
59  unsigned, std::list<std::list<dram_req_t *>::iterator> >[m_config->nbk];
60  m_last_write_row =
61  new std::list<std::list<dram_req_t *>::iterator> *[m_config->nbk];
62 
63  for (unsigned i = 0; i < m_config->nbk; i++) {
64  m_write_queue[i].clear();
65  m_write_bins[i].clear();
66  m_last_write_row[i] = NULL;
67  }
68  }
69  m_mode = READ_MODE;
70 }

72 void frfcfs_scheduler::add_req(dram_req_t *req) {
73  if (m_config->seperate_write_queue_enabled && req->data->is_write()) {
74  assert(m_num_write_pending < m_config->gpgpu_frfcfs_dram_write_queue_size);
75  m_num_write_pending++;
76  m_write_queue[req->bk].push_front(req);
77  std::list<dram_req_t *>::iterator ptr = m_write_queue[req->bk].begin();
78  m_write_bins[req->bk][req->row].push_front(ptr); // newest reqs to the
79  // front
80  } else {
81  assert(m_num_pending < m_config->gpgpu_frfcfs_dram_sched_queue_size);
82  m_num_pending++;
83  m_queue[req->bk].push_front(req);
84  std::list<dram_req_t *>::iterator ptr = m_queue[req->bk].begin();
85  m_bins[req->bk][req->row].push_front(ptr); // newest reqs to the front
86  }
87 }

89 void frfcfs_scheduler::data_collection(unsigned int bank) {
90  if (m_dram->m_gpu->gpu_sim_cycle > row_service_timestamp[bank]) {
91  curr_row_service_time[bank] =
92  m_dram->m_gpu->gpu_sim_cycle - row_service_timestamp[bank];
93  if (curr_row_service_time[bank] >
94  m_stats->max_servicetime2samerow[m_dram->id][bank])
95  m_stats->max_servicetime2samerow[m_dram->id][bank] =
96  curr_row_service_time[bank];
97  }
98  curr_row_service_time[bank] = 0;
99  row_service_timestamp[bank] = m_dram->m_gpu->gpu_sim_cycle;
100  if (m_stats->concurrent_row_access[m_dram->id][bank] >
101  m_stats->max_conc_access2samerow[m_dram->id][bank]) {
102  m_stats->max_conc_access2samerow[m_dram->id][bank] =
103  m_stats->concurrent_row_access[m_dram->id][bank];
104  }
105  m_stats->concurrent_row_access[m_dram->id][bank] = 0;
106  m_stats->num_activates[m_dram->id][bank]++;
107 }

109 dram_req_t *frfcfs_scheduler::schedule(unsigned bank, unsigned curr_row) {
110  // row
111  bool rowhit = true;
112  std::list<dram_req_t *> *m_current_queue = m_queue;
113  std::map<unsigned, std::list<std::list<dram_req_t *>::iterator> >
114  *m_current_bins = m_bins;
115  std::list<std::list<dram_req_t *>::iterator> **m_current_last_row =
116  m_last_row;
117 
118  if (m_config->seperate_write_queue_enabled) {
119  if (m_mode == READ_MODE &&
120  ((m_num_write_pending >= m_config->write_high_watermark)
121  // || (m_queue[bank].empty() && !m_write_queue[bank].empty())
122  )) {
123  m_mode = WRITE_MODE;
124  } else if (m_mode == WRITE_MODE &&
125  ((m_num_write_pending < m_config->write_low_watermark)
126  // || (!m_queue[bank].empty() && m_write_queue[bank].empty())
127  )) {
128  m_mode = READ_MODE;
129  }
130  }
131 
132  if (m_mode == WRITE_MODE) {
133  m_current_queue = m_write_queue;
134  m_current_bins = m_write_bins;
135  m_current_last_row = m_last_write_row;
136  }
137 
138  if (m_current_last_row[bank] == NULL) {
139  if (m_current_queue[bank].empty()) return NULL;
140 
141  std::map<unsigned, std::list<std::list<dram_req_t *>::iterator> >::iterator
142  bin_ptr = m_current_bins[bank].find(curr_row);
143  if (bin_ptr == m_current_bins[bank].end()) {
144  dram_req_t *req = m_current_queue[bank].back();
145  bin_ptr = m_current_bins[bank].find(req->row);
146  assert(bin_ptr !=
147  m_current_bins[bank].end()); // where did the request go???
148  m_current_last_row[bank] = &(bin_ptr->second);
149  data_collection(bank);
150  rowhit = false;
151  } else {
152  m_current_last_row[bank] = &(bin_ptr->second);
153  rowhit = true;
154  }
155  }
156  std::list<dram_req_t *>::iterator next = m_current_last_row[bank]->back();
157  dram_req_t *req = (*next);
158 
159  // rowblp stats
160  m_dram->access_num++;
161  bool is_write = req->data->is_write();
162  if (is_write)
163  m_dram->write_num++;
164  else
165  m_dram->read_num++;
166 
167  if (rowhit) {
168  m_dram->hits_num++;
169  if (is_write)
170  m_dram->hits_write_num++;
171  else
172  m_dram->hits_read_num++;
173  }
174 
175  m_stats->concurrent_row_access[m_dram->id][bank]++;
176  m_stats->row_access[m_dram->id][bank]++;
177  m_current_last_row[bank]->pop_back();
178 
179  m_current_queue[bank].erase(next);
180  if (m_current_last_row[bank]->empty()) {
181  m_current_bins[bank].erase(req->row);
182  m_current_last_row[bank] = NULL;
183  }
184 #ifdef DEBUG_FAST_IDEAL_SCHED
185  if (req)
186  printf("%08u : DRAM(%u) scheduling memory request to bank=%u, row=%u\n",
187  (unsigned)gpu_sim_cycle, m_dram->id, req->bk, req->row);
188 #endif
189 
190  if (m_config->seperate_write_queue_enabled && req->data->is_write()) {
191  assert(req != NULL && m_num_write_pending != 0);
192  m_num_write_pending--;
193  } else {
194  assert(req != NULL && m_num_pending != 0);
195  m_num_pending--;
196  }
197 
198  return req;
199 }

207 void dram_t::scheduler_frfcfs() {
208  unsigned mrq_latency;
209  frfcfs_scheduler *sched = m_frfcfs_scheduler;
210  while (!mrqq->empty()) {
211  dram_req_t *req = mrqq->pop();
212 
213  // Power stats
214  // if(req->data->get_type() != READ_REPLY && req->data->get_type() !=
215  // WRITE_ACK)
216  m_stats->total_n_access++;
217 
218  if (req->data->get_type() == WRITE_REQUEST) {
219  m_stats->total_n_writes++;
220  } else if (req->data->get_type() == READ_REQUEST) {
221  m_stats->total_n_reads++;
222  }
223 
224  req->data->set_status(IN_PARTITION_MC_INPUT_QUEUE,
225  m_gpu->gpu_sim_cycle + m_gpu->gpu_tot_sim_cycle);
226  sched->add_req(req);
227  }
228 
229  dram_req_t *req;
230  unsigned i;
231  for (i = 0; i < m_config->nbk; i++) {
232  unsigned b = (i + prio) % m_config->nbk;
233  if (!bk[b]->mrq) {
234  req = sched->schedule(b, bk[b]->curr_row);
235 
236  if (req) {
237  req->data->set_status(IN_PARTITION_MC_BANK_ARB_QUEUE,
238  m_gpu->gpu_sim_cycle + m_gpu->gpu_tot_sim_cycle);
239  prio = (prio + 1) % m_config->nbk;
240  bk[b]->mrq = req;
241  if (m_config->gpgpu_memlatency_stat) {
242  mrq_latency = m_gpu->gpu_sim_cycle + m_gpu->gpu_tot_sim_cycle -
243  bk[b]->mrq->timestamp;
244  m_stats->tot_mrq_latency += mrq_latency;
245  m_stats->tot_mrq_num++;
246  bk[b]->mrq->timestamp =
247  m_gpu->gpu_tot_sim_cycle + m_gpu->gpu_sim_cycle;
248  m_stats->mrq_lat_table[LOGB2(mrq_latency)]++;
249  if (mrq_latency > m_stats->max_mrq_latency) {
250  m_stats->max_mrq_latency = mrq_latency;
251  }
252  }
253 
254  break;
255  }
256  }
257  }
