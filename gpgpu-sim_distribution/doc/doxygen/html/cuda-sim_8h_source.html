66 class functionalCoreSim : public core_t {
67  public:
68  functionalCoreSim(kernel_info_t *kernel, gpgpu_sim *g, unsigned warp_size)
69  : core_t(g, kernel, warp_size, kernel->threads_per_cta()) {
70  m_warpAtBarrier = new bool[m_warp_count];
71  m_liveThreadCount = new unsigned[m_warp_count];
72  }
73  virtual ~functionalCoreSim() {
74  warp_exit(0);
75  delete[] m_liveThreadCount;
76  delete[] m_warpAtBarrier;
77  }
79  void execute(int inst_count, unsigned ctaid_cp);
80  virtual void warp_exit(unsigned warp_id);
81  virtual bool warp_waiting_at_barrier(unsigned warp_id) const {
82  return (m_warpAtBarrier[warp_id] || !(m_liveThreadCount[warp_id] > 0));
83  }
84 
85  private:
86  void executeWarp(unsigned, bool &, bool &);
87  // initializes threads in the CTA block which we are executing
88  void initializeCTA(unsigned ctaid_cp);
89  virtual void checkExecutionStatusAndUpdate(warp_inst_t &inst, unsigned t,
90  unsigned tid) {
91  if (m_thread[tid] == NULL || m_thread[tid]->is_done()) {
92  m_liveThreadCount[tid / m_warp_size]--;
93  }
94  }
95 
96  // lunches the stack and set the threads count
97  void createWarp(unsigned warpId);
98 
99  // each warp live thread count and barrier indicator
100  unsigned *m_liveThreadCount;
101  bool *m_warpAtBarrier;
102 };
