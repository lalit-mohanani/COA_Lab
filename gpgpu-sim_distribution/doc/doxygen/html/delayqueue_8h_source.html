45 template <class T>
46 class fifo_pipeline {
47  public:
48  fifo_pipeline(const char* nm, unsigned int minlen, unsigned int maxlen) {
49  assert(maxlen);
50  m_name = nm;
51  m_min_len = minlen;
52  m_max_len = maxlen;
53  m_length = 0;
54  m_n_element = 0;
55  m_head = NULL;
56  m_tail = NULL;
57  for (unsigned i = 0; i < m_min_len; i++) push(NULL);
58  }
59 
60  ~fifo_pipeline() {
61  while (m_head) {
62  m_tail = m_head;
63  m_head = m_head->m_next;
64  delete m_tail;
65  }
66  }
67 
68  void push(T* data) {
69  assert(m_length < m_max_len);
70  if (m_head) {
71  if (m_tail->m_data || m_length < m_min_len) {
72  m_tail->m_next = new fifo_data<T>();
73  m_tail = m_tail->m_next;
74  m_length++;
75  m_n_element++;
76  }
77  } else {
78  m_head = m_tail = new fifo_data<T>();
79  m_length++;
80  m_n_element++;
81  }
82  m_tail->m_next = NULL;
83  m_tail->m_data = data;
84  }
85 
86  T* pop() {
87  fifo_data<T>* next;
88  T* data;
89  if (m_head) {
90  next = m_head->m_next;
91  data = m_head->m_data;
92  if (m_head == m_tail) {
93  assert(next == NULL);
94  m_tail = NULL;
95  }
96  delete m_head;
97  m_head = next;
98  m_length--;
99  if (m_length == 0) {
100  assert(m_head == NULL);
101  m_tail = m_head;
102  }
103  m_n_element--;
104  if (m_min_len && m_length < m_min_len) {
105  push(NULL);
106  m_n_element--; // uncount NULL elements inserted to create delays
107  }
108  } else {
109  data = NULL;
110  }
111  return data;
112  }
113 
114  T* top() const {
115  if (m_head) {
116  return m_head->m_data;
117  } else {
118  return NULL;
119  }
120  }

122  void set_min_length(unsigned int new_min_len) {
123  if (new_min_len == m_min_len) return;
124 
125  if (new_min_len > m_min_len) {
126  m_min_len = new_min_len;
127  while (m_length < m_min_len) {
128  push(NULL);
129  m_n_element--; // uncount NULL elements inserted to create delays
130  }
131  } else {
132  // in this branch imply that the original min_len is larger then 0
133  // ie. head != 0
134  assert(m_head);
135  m_min_len = new_min_len;
136  while ((m_length > m_min_len) && (m_tail->m_data == 0)) {
137  fifo_data<T>* iter;
138  iter = m_head;
139  while (iter && (iter->m_next != m_tail)) iter = iter->m_next;
140  if (!iter) {
141  // there is only one node, and that node is empty
142  assert(m_head->m_data == 0);
143  pop();
144  } else {
145  // there are more than one node, and tail node is empty
146  assert(iter->m_next == m_tail);
147  delete m_tail;
148  m_tail = iter;
149  m_tail->m_next = 0;
150  m_length--;
151  }
152  }
153  }
154  }
