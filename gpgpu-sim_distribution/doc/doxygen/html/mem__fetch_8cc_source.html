86 void mem_fetch::print(FILE *fp, bool print_inst) const {

130 unsigned mem_fetch::get_num_flits(bool simt_to_mem) {
131  unsigned sz = 0;
132  // If atomic, write going to memory, or read coming back from memory, size =
133  // ctrl + data. Else, only ctrl
134  if (isatomic() || (simt_to_mem && get_is_write()) ||
135  !(simt_to_mem || get_is_write()))
136  sz = size();
137  else
138  sz = get_ctrl_size();
139 
140  return (sz / icnt_flit_size) + ((sz % icnt_flit_size) ? 1 : 0);
141 }
