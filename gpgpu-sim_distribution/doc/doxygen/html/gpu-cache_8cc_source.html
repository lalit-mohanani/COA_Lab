76 unsigned cache_config::hash_function(new_addr_type addr, unsigned m_nset,
77  unsigned m_line_sz_log2,
78  unsigned m_nset_log2,
79  unsigned m_index_function) const {
80  unsigned set_index = 0;
81 
82  switch (m_index_function) {
83  case FERMI_HASH_SET_FUNCTION: {
84  /*
85  * Set Indexing function from "A Detailed GPU Cache Model Based on Reuse
86  * Distance Theory" Cedric Nugteren et al. HPCA 2014
87  */
88  unsigned lower_xor = 0;
89  unsigned upper_xor = 0;
90 
91  if (m_nset == 32 || m_nset == 64) {
92  // Lower xor value is bits 7-11
93  lower_xor = (addr >> m_line_sz_log2) & 0x1F;
94 
95  // Upper xor value is bits 13, 14, 15, 17, and 19
96  upper_xor = (addr & 0xE000) >> 13; // Bits 13, 14, 15
97  upper_xor |= (addr & 0x20000) >> 14; // Bit 17
98  upper_xor |= (addr & 0x80000) >> 15; // Bit 19
99 
100  set_index = (lower_xor ^ upper_xor);
101 
102  // 48KB cache prepends the set_index with bit 12
103  if (m_nset == 64) set_index |= (addr & 0x1000) >> 7;
104 
105  } else { /* Else incorrect number of sets for the hashing function */
106  assert(
107  "\nGPGPU-Sim cache configuration error: The number of sets should "
108  "be "
109  "32 or 64 for the hashing set index function.\n" &&
110  0);
111  }
112  break;
113  }
114 
115  case BITWISE_XORING_FUNCTION: {
116  new_addr_type higher_bits = addr >> (m_line_sz_log2 + m_nset_log2);
117  unsigned index = (addr >> m_line_sz_log2) & (m_nset - 1);
118  set_index = bitwise_hash_function(higher_bits, index, m_nset);
119  break;
120  }
121  case HASH_IPOLY_FUNCTION: {
122  new_addr_type higher_bits = addr >> (m_line_sz_log2 + m_nset_log2);
123  unsigned index = (addr >> m_line_sz_log2) & (m_nset - 1);
124  set_index = ipoly_hash_function(higher_bits, index, m_nset);
125  break;
126  }
127  case CUSTOM_SET_FUNCTION: {
128  /* No custom set function implemented */
129  break;
130  }
131 
132  case LINEAR_SET_FUNCTION: {
133  set_index = (addr >> m_line_sz_log2) & (m_nset - 1);
134  break;
135  }
136 
137  default: {
138  assert("\nUndefined set index function.\n" && 0);
139  break;
140  }
141  }
142 
143  // Linear function selected or custom set index function not implemented
144  assert((set_index < m_nset) &&
145  "\nError: Set index out of bounds. This is caused by "
146  "an incorrect or unimplemented custom set index function.\n");
147 
148  return set_index;
149 }

240 enum cache_request_status tag_array::probe(new_addr_type addr, unsigned &idx,
241  mem_access_sector_mask_t mask,
242  bool probe_mode,
243  mem_fetch *mf) const {
244  // assert( m_config.m_write_policy == READ_ONLY );
245  unsigned set_index = m_config.set_index(addr);
246  new_addr_type tag = m_config.tag(addr);
247 
248  unsigned invalid_line = (unsigned)-1;
249  unsigned valid_line = (unsigned)-1;
250  unsigned long long valid_timestamp = (unsigned)-1;
251 
252  bool all_reserved = true;
253 
254  // check for hit or pending hit
255  for (unsigned way = 0; way < m_config.m_assoc; way++) {
256  unsigned index = set_index * m_config.m_assoc + way;
257  cache_block_t *line = m_lines[index];
258  if (line->m_tag == tag) {
259  if (line->get_status(mask) == RESERVED) {
260  idx = index;
261  return HIT_RESERVED;
262  } else if (line->get_status(mask) == VALID) {
263  idx = index;
264  return HIT;
265  } else if (line->get_status(mask) == MODIFIED) {
266  if (line->is_readable(mask)) {
267  idx = index;
268  return HIT;
269  } else {
270  idx = index;
271  return SECTOR_MISS;
272  }
273 
274  } else if (line->is_valid_line() && line->get_status(mask) == INVALID) {
275  idx = index;
276  return SECTOR_MISS;
277  } else {
278  assert(line->get_status(mask) == INVALID);
279  }
280  }
281  if (!line->is_reserved_line()) {
282  all_reserved = false;
283  if (line->is_invalid_line()) {
284  invalid_line = index;
285  } else {
286  // valid line : keep track of most appropriate replacement candidate
287  if (m_config.m_replacement_policy == LRU) {
288  if (line->get_last_access_time() < valid_timestamp) {
289  valid_timestamp = line->get_last_access_time();
290  valid_line = index;
291  }
292  } else if (m_config.m_replacement_policy == FIFO) {
293  if (line->get_alloc_time() < valid_timestamp) {
294  valid_timestamp = line->get_alloc_time();
295  valid_line = index;
296  }
297  }
298  }
299  }
300  }
301  if (all_reserved) {
302  assert(m_config.m_alloc_policy == ON_MISS);
303  return RESERVATION_FAIL; // miss and not enough space in cache to allocate
304  // on miss
305  }
306 
307  if (invalid_line != (unsigned)-1) {
308  idx = invalid_line;
309  } else if (valid_line != (unsigned)-1) {
310  idx = valid_line;
311  } else
312  abort(); // if an unreserved block exists, it is either invalid or
313  // replaceable
314 
315  if (probe_mode && m_config.is_streaming()) {
316  line_table::const_iterator i =
317  pending_lines.find(m_config.block_addr(addr));
318  assert(mf);
319  if (!mf->is_write() && i != pending_lines.end()) {
320  if (i->second != mf->get_inst().get_uid()) return SECTOR_MISS;
321  }
322  }
323 
324  return MISS;
325 }

336 enum cache_request_status tag_array::access(new_addr_type addr, unsigned time,
337  unsigned &idx, bool &wb,
338  evicted_block_info &evicted,
339  mem_fetch *mf) {
340  m_access++;
341  is_used = true;
342  shader_cache_access_log(m_core_id, m_type_id, 0); // log accesses to cache
343  enum cache_request_status status = probe(addr, idx, mf);
344  switch (status) {
345  case HIT_RESERVED:
346  m_pending_hit++;
347  case HIT:
348  m_lines[idx]->set_last_access_time(time, mf->get_access_sector_mask());
349  break;
350  case MISS:
351  m_miss++;
352  shader_cache_access_log(m_core_id, m_type_id, 1); // log cache misses
353  if (m_config.m_alloc_policy == ON_MISS) {
354  if (m_lines[idx]->is_modified_line()) {
355  wb = true;
356  evicted.set_info(m_lines[idx]->m_block_addr,
357  m_lines[idx]->get_modified_size());
358  }
359  m_lines[idx]->allocate(m_config.tag(addr), m_config.block_addr(addr),
360  time, mf->get_access_sector_mask());
361  }
362  break;
363  case SECTOR_MISS:
364  assert(m_config.m_cache_type == SECTOR);
365  m_sector_miss++;
366  shader_cache_access_log(m_core_id, m_type_id, 1); // log cache misses
367  if (m_config.m_alloc_policy == ON_MISS) {
368  ((sector_cache_block *)m_lines[idx])
369  ->allocate_sector(time, mf->get_access_sector_mask());
370  }
371  break;
372  case RESERVATION_FAIL:
373  m_res_fail++;
374  shader_cache_access_log(m_core_id, m_type_id, 1); // log cache misses
375  break;
376  default:
377  fprintf(stderr,
378  "tag_array::access - Error: Unknown"
379  "cache_request_status %d\n",
380  status);
381  abort();
382  }
383  return status;
384 }

955 void baseline_cache::bandwidth_management::use_data_port(
956  mem_fetch *mf, enum cache_request_status outcome,
957  const std::list<cache_event> &events) {
958  unsigned data_size = mf->get_data_size();
959  unsigned port_width = m_config.m_data_port_width;
960  switch (outcome) {
961  case HIT: {
962  unsigned data_cycles =
963  data_size / port_width + ((data_size % port_width > 0) ? 1 : 0);
964  m_data_port_occupied_cycles += data_cycles;
965  } break;
966  case HIT_RESERVED:
967  case MISS: {
968  // the data array is accessed to read out the entire line for write-back
969  // in case of sector cache we need to write bank only the modified sectors
970  cache_event ev(WRITE_BACK_REQUEST_SENT);
971  if (was_writeback_sent(events, ev)) {
972  unsigned data_cycles = ev.m_evicted_block.m_modified_size / port_width;
973  m_data_port_occupied_cycles += data_cycles;
974  }
975  } break;
976  case SECTOR_MISS:
977  case RESERVATION_FAIL:
978  // Does not consume any port bandwidth
979  break;
980  default:
981  assert(0);
982  break;
983  }

1040
