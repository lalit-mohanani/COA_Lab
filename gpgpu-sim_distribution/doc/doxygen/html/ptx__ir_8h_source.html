122 struct type_info_key_compare {
123  bool operator()(const type_info_key &a, const type_info_key &b) const {
124  assert(a.m_init && b.m_init);
125  if (a.m_space_spec < b.m_space_spec) return true;
126  if (a.m_scalar_type_spec < b.m_scalar_type_spec) return true;
127  if (a.m_vector_spec < b.m_vector_spec) return true;
128  if (a.m_alignment_spec < b.m_alignment_spec) return true;
129  if (a.m_extern_spec < b.m_extern_spec) return true;
130  if (a.m_array_dim < b.m_array_dim) return true;
131  if (a.m_is_function < b.m_is_function) return true;
132 
133  return false;
134  }
