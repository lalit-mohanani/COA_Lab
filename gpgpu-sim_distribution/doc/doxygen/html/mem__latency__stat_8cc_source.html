188 // record the total latency
189 unsigned memory_stats_t::memlatstat_done(mem_fetch *mf) {
190  unsigned mf_latency;
191  mf_latency =
192  (m_gpu->gpu_sim_cycle + m_gpu->gpu_tot_sim_cycle) - mf->get_timestamp();
193  mf_num_lat_pw++;
194  mf_tot_lat_pw += mf_latency;
195  unsigned idx = LOGB2(mf_latency);
196  assert(idx < 32);
197  mf_lat_table[idx]++;
198  shader_mem_lat_log(mf->get_sid(), mf_latency);
199  mf_total_lat_table[mf->get_tlx_addr().chip][mf->get_tlx_addr().bk] +=
200  mf_latency;
201  if (mf_latency > max_mf_latency) max_mf_latency = mf_latency;
202  return mf_latency;
203 }

205 void memory_stats_t::memlatstat_read_done(mem_fetch *mf) {
206  if (m_memory_config->gpgpu_memlatency_stat) {
207  unsigned mf_latency = memlatstat_done(mf);
208  if (mf_latency >
209  mf_max_lat_table[mf->get_tlx_addr().chip][mf->get_tlx_addr().bk])
210  mf_max_lat_table[mf->get_tlx_addr().chip][mf->get_tlx_addr().bk] =
211  mf_latency;
212  unsigned icnt2sh_latency;
213  icnt2sh_latency = (m_gpu->gpu_tot_sim_cycle + m_gpu->gpu_sim_cycle) -
214  mf->get_return_timestamp();
215  tot_icnt2sh_latency += icnt2sh_latency;
216  icnt2sh_lat_table[LOGB2(icnt2sh_latency)]++;
217  if (icnt2sh_latency > max_icnt2sh_latency)
218  max_icnt2sh_latency = icnt2sh_latency;
219  }
220 }

222 void memory_stats_t::memlatstat_dram_access(mem_fetch *mf) {
223  unsigned dram_id = mf->get_tlx_addr().chip;
224  unsigned bank = mf->get_tlx_addr().bk;
225  if (m_memory_config->gpgpu_memlatency_stat) {
226  if (mf->get_is_write()) {
227  if (mf->get_sid() < m_n_shader) { // do not count L2_writebacks here
228  bankwrites[mf->get_sid()][dram_id][bank]++;
229  shader_mem_acc_log(mf->get_sid(), dram_id, bank, 'w');
230  }
231  totalbankwrites[dram_id][bank] +=
232  ceil(mf->get_data_size() / m_memory_config->dram_atom_size);
233  } else {
234  bankreads[mf->get_sid()][dram_id][bank]++;
235  shader_mem_acc_log(mf->get_sid(), dram_id, bank, 'r');
236  totalbankreads[dram_id][bank] +=
237  ceil(mf->get_data_size() / m_memory_config->dram_atom_size);
238  }
239  mem_access_type_stats[mf->get_access_type()][dram_id][bank] +=
240  ceil(mf->get_data_size() / m_memory_config->dram_atom_size);
241  }
242 
243  if (mf->get_pc() != (unsigned)-1)
244  m_gpu->gpgpu_ctx->stats->ptx_file_line_stats_add_dram_traffic(
245  mf->get_pc(), mf->get_data_size());
246 }

249  if (m_memory_config->gpgpu_memlatency_stat) {

251  icnt2mem_latency =

253  tot_icnt2mem_latency += icnt2mem_latency;
254  icnt2mem_lat_table[LOGB2(icnt2mem_latency)]++;
255  if (icnt2mem_latency > max_icnt2mem_latency)
256  max_icnt2mem_latency = icnt2mem_latency;
257  }

261  if (mf_num_lat_pw && m_memory_config->gpgpu_memlatency_stat) {

263  mf_total_lat += mf_tot_lat_pw;
