104 memory_partition_unit::arbitration_metadata::arbitration_metadata(
105  const memory_config *config)
106  : m_last_borrower(config->m_n_sub_partition_per_memory_channel - 1),
107  m_private_credit(config->m_n_sub_partition_per_memory_channel, 0),
108  m_shared_credit(0) {
109  // each sub partition get at least 1 credit for forward progress
110  // the rest is shared among with other partitions
111  m_private_credit_limit = 1;
112  m_shared_credit_limit = config->gpgpu_frfcfs_dram_sched_queue_size +
113  config->gpgpu_dram_return_queue_size -
114  (config->m_n_sub_partition_per_memory_channel - 1);
115  if (config->seperate_write_queue_enabled)
116  m_shared_credit_limit += config->gpgpu_frfcfs_dram_write_queue_size;
117  if (config->gpgpu_frfcfs_dram_sched_queue_size == 0 or
118  config->gpgpu_dram_return_queue_size == 0) {
119  m_shared_credit_limit =
120  0; // no limit if either of the queue has no limit in size
121  }
122  assert(m_shared_credit_limit >= 0);
123 }

125 bool memory_partition_unit::arbitration_metadata::has_credits(
126  int inner_sub_partition_id) const {
127  int spid = inner_sub_partition_id;
128  if (m_private_credit[spid] < m_private_credit_limit) {
129  return true;
130  } else if (m_shared_credit_limit == 0 ||
131  m_shared_credit < m_shared_credit_limit) {
132  return true;
133  } else {
134  return false;
135  }
136 }

138 void memory_partition_unit::arbitration_metadata::borrow_credit(
139  int inner_sub_partition_id) {
140  int spid = inner_sub_partition_id;
141  if (m_private_credit[spid] < m_private_credit_limit) {
142  m_private_credit[spid] += 1;
143  } else if (m_shared_credit_limit == 0 ||
144  m_shared_credit < m_shared_credit_limit) {
145  m_shared_credit += 1;
146  } else {
147  assert(0 && "DRAM arbitration error: Borrowing from depleted credit!");
148  }
149  m_last_borrower = spid;
150 }

152 void memory_partition_unit::arbitration_metadata::return_credit(
153  int inner_sub_partition_id) {
154  int spid = inner_sub_partition_id;
155  if (m_private_credit[spid] > 0) {
156  m_private_credit[spid] -= 1;
157  } else {
158  m_shared_credit -= 1;
159  }
160  assert((m_shared_credit >= 0) &&
161  "DRAM arbitration error: Returning more than available credits!");
162 }

208  spid, (sub_partition_contention) ? 'T' : 'F',

220 void memory_partition_unit::simple_dram_model_cycle() {
221  // pop completed memory request from dram and push it to dram-to-L2 queue
222  // of the original sub partition
223  if (!m_dram_latency_queue.empty() &&
224  ((m_gpu->gpu_sim_cycle + m_gpu->gpu_tot_sim_cycle) >=
225  m_dram_latency_queue.front().ready_cycle)) {
226  mem_fetch *mf_return = m_dram_latency_queue.front().req;
227  if (mf_return->get_access_type() != L1_WRBK_ACC &&
228  mf_return->get_access_type() != L2_WRBK_ACC) {
229  mf_return->set_reply();
230 
231  unsigned dest_global_spid = mf_return->get_sub_partition_id();
232  int dest_spid = global_sub_partition_id_to_local_id(dest_global_spid);
233  assert(m_sub_partition[dest_spid]->get_id() == dest_global_spid);
234  if (!m_sub_partition[dest_spid]->dram_L2_queue_full()) {
235  if (mf_return->get_access_type() == L1_WRBK_ACC) {
236  m_sub_partition[dest_spid]->set_done(mf_return);
237  delete mf_return;
238  } else {
239  m_sub_partition[dest_spid]->dram_L2_queue_push(mf_return);
240  mf_return->set_status(
241  IN_PARTITION_DRAM_TO_L2_QUEUE,
242  m_gpu->gpu_sim_cycle + m_gpu->gpu_tot_sim_cycle);
243  m_arbitration_metadata.return_credit(dest_spid);
244  MEMPART_DPRINTF(
245  "mem_fetch request %p return from dram to sub partition %d\n",
246  mf_return, dest_spid);
247  }
248  m_dram_latency_queue.pop_front();
249  }
250 
251  } else {
252  this->set_done(mf_return);
253  delete mf_return;
254  m_dram_latency_queue.pop_front();
255  }
256  }
257 
258  // mem_fetch *mf = m_sub_partition[spid]->L2_dram_queue_top();
259  // if( !m_dram->full(mf->is_write()) ) {
260  // L2->DRAM queue to DRAM latency queue
261  // Arbitrate among multiple L2 subpartitions
262  int last_issued_partition = m_arbitration_metadata.last_borrower();
263  for (unsigned p = 0; p < m_config->m_n_sub_partition_per_memory_channel;
264  p++) {
265  int spid = (p + last_issued_partition + 1) %
266  m_config->m_n_sub_partition_per_memory_channel;
267  if (!m_sub_partition[spid]->L2_dram_queue_empty() &&
268  can_issue_to_dram(spid)) {
269  mem_fetch *mf = m_sub_partition[spid]->L2_dram_queue_top();
270  if (m_dram->full(mf->is_write())) break;
271 
272  m_sub_partition[spid]->L2_dram_queue_pop();
273  MEMPART_DPRINTF(
274  "Issue mem_fetch request %p from sub partition %d to dram\n", mf,
275  spid);
276  dram_delay_t d;
277  d.req = mf;
278  d.ready_cycle = m_gpu->gpu_sim_cycle + m_gpu->gpu_tot_sim_cycle +
279  m_config->dram_latency;
280  m_dram_latency_queue.push_back(d);
281  mf->set_status(IN_PARTITION_DRAM_LATENCY_QUEUE,
282  m_gpu->gpu_sim_cycle + m_gpu->gpu_tot_sim_cycle);
283  m_arbitration_metadata.borrow_credit(spid);
284  break; // the DRAM should only accept one request per cycle
285  }
286  }

290 void memory_partition_unit::dram_cycle() {
291  // pop completed memory request from dram and push it to dram-to-L2 queue
292  // of the original sub partition
293  mem_fetch *mf_return = m_dram->return_queue_top();
294  if (mf_return) {
295  unsigned dest_global_spid = mf_return->get_sub_partition_id();
296  int dest_spid = global_sub_partition_id_to_local_id(dest_global_spid);
297  assert(m_sub_partition[dest_spid]->get_id() == dest_global_spid);
298  if (!m_sub_partition[dest_spid]->dram_L2_queue_full()) {
299  if (mf_return->get_access_type() == L1_WRBK_ACC) {
300  m_sub_partition[dest_spid]->set_done(mf_return);
301  delete mf_return;
302  } else {
303  m_sub_partition[dest_spid]->dram_L2_queue_push(mf_return);
304  mf_return->set_status(IN_PARTITION_DRAM_TO_L2_QUEUE,
305  m_gpu->gpu_sim_cycle + m_gpu->gpu_tot_sim_cycle);
306  m_arbitration_metadata.return_credit(dest_spid);
307  MEMPART_DPRINTF(
308  "mem_fetch request %p return from dram to sub partition %d\n",
309  mf_return, dest_spid);
310  }
311  m_dram->return_queue_pop();
312  }
313  } else {
314  m_dram->return_queue_pop();
315  }
316 
317  m_dram->cycle();
318  m_dram->dram_log(SAMPLELOG);
319 
320  // mem_fetch *mf = m_sub_partition[spid]->L2_dram_queue_top();
321  // if( !m_dram->full(mf->is_write()) ) {
322  // L2->DRAM queue to DRAM latency queue
323  // Arbitrate among multiple L2 subpartitions
324  int last_issued_partition = m_arbitration_metadata.last_borrower();
325  for (unsigned p = 0; p < m_config->m_n_sub_partition_per_memory_channel;
326  p++) {
327  int spid = (p + last_issued_partition + 1) %
328  m_config->m_n_sub_partition_per_memory_channel;
329  if (!m_sub_partition[spid]->L2_dram_queue_empty() &&
330  can_issue_to_dram(spid)) {
331  mem_fetch *mf = m_sub_partition[spid]->L2_dram_queue_top();
332  if (m_dram->full(mf->is_write())) break;
333 
334  m_sub_partition[spid]->L2_dram_queue_pop();
335  MEMPART_DPRINTF(
336  "Issue mem_fetch request %p from sub partition %d to dram\n", mf,
337  spid);
338  dram_delay_t d;
339  d.req = mf;
340  d.ready_cycle = m_gpu->gpu_sim_cycle + m_gpu->gpu_tot_sim_cycle +
341  m_config->dram_latency;
342  m_dram_latency_queue.push_back(d);
343  mf->set_status(IN_PARTITION_DRAM_LATENCY_QUEUE,
344  m_gpu->gpu_sim_cycle + m_gpu->gpu_tot_sim_cycle);
345  m_arbitration_metadata.borrow_credit(spid);
346  break; // the DRAM should only accept one request per cycle
347  }
348  }
349  //}
350 
351  // DRAM latency queue
352  if (!m_dram_latency_queue.empty() &&
353  ((m_gpu->gpu_sim_cycle + m_gpu->gpu_tot_sim_cycle) >=
354  m_dram_latency_queue.front().ready_cycle) &&
355  !m_dram->full(m_dram_latency_queue.front().req->is_write())) {
356  mem_fetch *mf = m_dram_latency_queue.front().req;
357  m_dram_latency_queue.pop_front();
358  m_dram->push(mf);
359  }

448 void memory_sub_partition::cache_cycle(unsigned cycle) {
449  // L2 fill responses
450  if (!m_config->m_L2_config.disabled()) {
451  if (m_L2cache->access_ready() && !m_L2_icnt_queue->full()) {
452  mem_fetch *mf = m_L2cache->next_access();
453  if (mf->get_access_type() !=
454  L2_WR_ALLOC_R) { // Don't pass write allocate read request back to
455  // upper level cache
456  mf->set_reply();
457  mf->set_status(IN_PARTITION_L2_TO_ICNT_QUEUE,
458  m_gpu->gpu_sim_cycle + m_gpu->gpu_tot_sim_cycle);
459  m_L2_icnt_queue->push(mf);
460  } else {
461  if (m_config->m_L2_config.m_write_alloc_policy == FETCH_ON_WRITE) {
462  mem_fetch *original_wr_mf = mf->get_original_wr_mf();
463  assert(original_wr_mf);
464  original_wr_mf->set_reply();
465  original_wr_mf->set_status(
466  IN_PARTITION_L2_TO_ICNT_QUEUE,
467  m_gpu->gpu_sim_cycle + m_gpu->gpu_tot_sim_cycle);
468  m_L2_icnt_queue->push(original_wr_mf);
469  }
470  m_request_tracker.erase(mf);
471  delete mf;
472  }
473  }
474  }
475 
476  // DRAM to L2 (texture) and icnt (not texture)
477  if (!m_dram_L2_queue->empty()) {
478  mem_fetch *mf = m_dram_L2_queue->top();
479  if (!m_config->m_L2_config.disabled() && m_L2cache->waiting_for_fill(mf)) {
480  if (m_L2cache->fill_port_free()) {
481  mf->set_status(IN_PARTITION_L2_FILL_QUEUE,
482  m_gpu->gpu_sim_cycle + m_gpu->gpu_tot_sim_cycle);
483  m_L2cache->fill(mf, m_gpu->gpu_sim_cycle + m_gpu->gpu_tot_sim_cycle +
484  m_memcpy_cycle_offset);
485  m_dram_L2_queue->pop();
486  }
487  } else if (!m_L2_icnt_queue->full()) {
488  if (mf->is_write() && mf->get_type() == WRITE_ACK)
489  mf->set_status(IN_PARTITION_L2_TO_ICNT_QUEUE,
490  m_gpu->gpu_sim_cycle + m_gpu->gpu_tot_sim_cycle);
491  m_L2_icnt_queue->push(mf);
492  m_dram_L2_queue->pop();
493  }
494  }
495 
496  // prior L2 misses inserted into m_L2_dram_queue here
497  if (!m_config->m_L2_config.disabled()) m_L2cache->cycle();
498 
499  // new L2 texture accesses and/or non-texture accesses
500  if (!m_L2_dram_queue->full() && !m_icnt_L2_queue->empty()) {
501  mem_fetch *mf = m_icnt_L2_queue->top();
502  if (!m_config->m_L2_config.disabled() &&
503  ((m_config->m_L2_texure_only && mf->istexture()) ||
504  (!m_config->m_L2_texure_only))) {
505  // L2 is enabled and access is for L2
506  bool output_full = m_L2_icnt_queue->full();
507  bool port_free = m_L2cache->data_port_free();
508  if (!output_full && port_free) {
509  std::list<cache_event> events;
510  enum cache_request_status status =
511  m_L2cache->access(mf->get_addr(), mf,
512  m_gpu->gpu_sim_cycle + m_gpu->gpu_tot_sim_cycle +
513  m_memcpy_cycle_offset,
514  events);
515  bool write_sent = was_write_sent(events);
516  bool read_sent = was_read_sent(events);
517  MEM_SUBPART_DPRINTF("Probing L2 cache Address=%llx, status=%u\n",
518  mf->get_addr(), status);
519 
520  if (status == HIT) {
521  if (!write_sent) {
522  // L2 cache replies
523  assert(!read_sent);
524  if (mf->get_access_type() == L1_WRBK_ACC) {
525  m_request_tracker.erase(mf);
526  delete mf;
527  } else {
528  mf->set_reply();
529  mf->set_status(IN_PARTITION_L2_TO_ICNT_QUEUE,
530  m_gpu->gpu_sim_cycle + m_gpu->gpu_tot_sim_cycle);
531  m_L2_icnt_queue->push(mf);
532  }
533  m_icnt_L2_queue->pop();
534  } else {
535  assert(write_sent);
536  m_icnt_L2_queue->pop();
537  }
538  } else if (status != RESERVATION_FAIL) {
539  if (mf->is_write() &&
540  (m_config->m_L2_config.m_write_alloc_policy == FETCH_ON_WRITE ||
541  m_config->m_L2_config.m_write_alloc_policy ==
542  LAZY_FETCH_ON_READ) &&
543  !was_writeallocate_sent(events)) {
544  mf->set_reply();
545  mf->set_status(IN_PARTITION_L2_TO_ICNT_QUEUE,
546  m_gpu->gpu_sim_cycle + m_gpu->gpu_tot_sim_cycle);
547  m_L2_icnt_queue->push(mf);
548  }
549  // L2 cache accepted request
550  m_icnt_L2_queue->pop();
551  } else {
552  assert(!write_sent);
553  assert(!read_sent);
554  // L2 cache lock-up: will try again next cycle
555  }
556  }
557  } else {
558  // L2 is disabled or non-texture access to texture-only L2
559  mf->set_status(IN_PARTITION_L2_TO_DRAM_QUEUE,
560  m_gpu->gpu_sim_cycle + m_gpu->gpu_tot_sim_cycle);
561  m_L2_dram_queue->push(mf);
562  m_icnt_L2_queue->pop();
563  }
564  }
565 
566  // ROP delay queue
567  if (!m_rop.empty() && (cycle >= m_rop.front().ready_cycle) &&
568  !m_icnt_L2_queue->full()) {
569  mem_fetch *mf = m_rop.front().req;
570  m_rop.pop();
571  m_icnt_L2_queue->push(mf);
572  mf->set_status(IN_PARTITION_ICNT_TO_L2_QUEUE,
573  m_gpu->gpu_sim_cycle + m_gpu->gpu_tot_sim_cycle);
574  }
575 }

694 std::vector<mem_fetch *>
695 memory_sub_partition::breakdown_request_to_sector_requests(mem_fetch *mf) {
696  std::vector<mem_fetch *> result;
697 
698  if (mf->get_data_size() == SECTOR_SIZE &&
699  mf->get_access_sector_mask().count() == 1) {
700  result.push_back(mf);
701  } else if (mf->get_data_size() == 128 || mf->get_data_size() == 64) {
702  // We only accept 32, 64 and 128 bytes reqs
703  unsigned start = 0, end = 0;
704  if (mf->get_data_size() == 128) {
705  start = 0;
706  end = 3;
707  } else if (mf->get_data_size() == 64 &&
708  mf->get_access_sector_mask().to_string() == "1100") {
709  start = 2;
710  end = 3;
711  } else if (mf->get_data_size() == 64 &&
712  mf->get_access_sector_mask().to_string() == "0011") {
713  start = 0;
714  end = 1;
715  } else if (mf->get_data_size() == 64 &&
716  (mf->get_access_sector_mask().to_string() == "1111" ||
717  mf->get_access_sector_mask().to_string() == "0000")) {
718  if (mf->get_addr() % 128 == 0) {
719  start = 0;
720  end = 1;
721  } else {
722  start = 2;
723  end = 3;
724  }
725  } else {
726  printf(
727  "Invalid sector received, address = 0x%06llx, sector mask = %s, data "
728  "size = %d",
729  mf->get_addr(), mf->get_access_sector_mask(), mf->get_data_size());
730  assert(0 && "Undefined sector mask is received");
731  }
732 
733  std::bitset<SECTOR_SIZE * SECTOR_CHUNCK_SIZE> byte_sector_mask;
734  byte_sector_mask.reset();
735  for (unsigned k = start * SECTOR_SIZE; k < SECTOR_SIZE; ++k)
736  byte_sector_mask.set(k);
737 
738  for (unsigned j = start, i = 0; j <= end; ++j, ++i) {
739  const mem_access_t *ma = new mem_access_t(
740  mf->get_access_type(), mf->get_addr() + SECTOR_SIZE * i, SECTOR_SIZE,
741  mf->is_write(), mf->get_access_warp_mask(),
742  mf->get_access_byte_mask() & byte_sector_mask,
743  std::bitset<SECTOR_CHUNCK_SIZE>().set(j), m_gpu->gpgpu_ctx);
744 
745  mem_fetch *n_mf =
746  new mem_fetch(*ma, NULL, mf->get_ctrl_size(), mf->get_wid(),
747  mf->get_sid(), mf->get_tpc(), mf->get_mem_config(),
748  m_gpu->gpu_tot_sim_cycle + m_gpu->gpu_sim_cycle, mf);
749 
750  result.push_back(n_mf);
751  byte_sector_mask <<= SECTOR_SIZE;
752  }
753  } else {
754  printf(
755  "Invalid sector received, address = 0x%06llx, sector mask = %d, byte "
756  "mask = , data size = %u",
757  mf->get_addr(), mf->get_access_sector_mask().count(),
758  mf->get_data_size());
759  assert(0 && "Undefined data size is received");
760  }
761 
762  return result;
763 }

765 void memory_sub_partition::push(mem_fetch *m_req, unsigned long long cycle) {
766  if (m_req) {
767  m_stats->memlatstat_icnt2mem_pop(m_req);
768  std::vector<mem_fetch *> reqs;
769  if (m_config->m_L2_config.m_cache_type == SECTOR)
770  reqs = breakdown_request_to_sector_requests(m_req);
771  else
772  reqs.push_back(m_req);
773 
774  for (unsigned i = 0; i < reqs.size(); ++i) {
775  mem_fetch *req = reqs[i];
776  m_request_tracker.insert(req);
777  if (req->istexture()) {
778  m_icnt_L2_queue->push(req);
779  req->set_status(IN_PARTITION_ICNT_TO_L2_QUEUE,
780  m_gpu->gpu_sim_cycle + m_gpu->gpu_tot_sim_cycle);
781  } else {
782  rop_delay_t r;
783  r.req = req;
784  r.ready_cycle = cycle + m_config->rop_latency;
785  m_rop.push(r);
786  req->set_status(IN_PARTITION_ROP_DELAY,
787  m_gpu->gpu_sim_cycle + m_gpu->gpu_tot_sim_cycle);
788  }
789  }
790  }
