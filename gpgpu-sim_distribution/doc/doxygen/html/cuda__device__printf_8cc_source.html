35 void my_cuda_printf(const char *fmtstr, const char *arg_list) {
36  FILE *fp = stdout;
37  unsigned i = 0, j = 0;
38  unsigned arg_offset = 0;
39  char buf[64];
40  bool in_fmt = false;
41  while (fmtstr[i]) {
42  char c = fmtstr[i++];
43  if (!in_fmt) {
44  if (c != '%') {
45  fprintf(fp, "%c", c);
46  } else {
47  in_fmt = true;
48  buf[0] = c;
49  j = 1;
50  }
51  } else {
52  if (!(c == 'u' || c == 'f' || c == 'd')) {
53  printf(
54  "GPGPU-Sim PTX: ERROR ** printf parsing support is limited to %%u, "
55  "%%f, %%d at present");
56  abort();
57  }
58  buf[j] = c;
59  buf[j + 1] = 0;
60  void *ptr = (void *)&arg_list[arg_offset];
61  // unsigned long long value = ((unsigned long long*)arg_list)[arg_offset];
62  if (c == 'u' || c == 'd') {
63  fprintf(fp, buf, *((unsigned long long *)ptr));
64  } else if (c == 'f') {
65  double tmp = *((double *)ptr);
66  fprintf(fp, buf, tmp);
67  }
68  arg_offset++;
69  in_fmt = false;
70  }
71  }
72 }

74 void gpgpusim_cuda_vprintf(const ptx_instruction *pI, ptx_thread_info *thread,
75  const function_info *target_func) {
76  char *fmtstr = NULL;
77  char *arg_list = NULL;
78  unsigned n_return = target_func->has_return();
79  unsigned n_args = target_func->num_args();
80  assert(n_args == 2);
81  for (unsigned arg = 0; arg < n_args; arg++) {
82  const operand_info &actual_param_op =
83  pI->operand_lookup(n_return + 1 + arg);
84  const symbol *formal_param = target_func->get_arg(arg);
85  unsigned size = formal_param->get_size_in_bytes();
86  assert(formal_param->is_param_local());
87  assert(actual_param_op.is_param_local());
88  addr_t from_addr = actual_param_op.get_symbol()->get_address();
89  unsigned long long buffer[1024];
90  assert(size < 1024 * sizeof(unsigned long long));
91  thread->m_local_mem->read(from_addr, size, buffer);
92  addr_t addr =
93  (addr_t)buffer[0]; // should be pointer to generic memory location
94  memory_space *mem = NULL;
95  memory_space_t space = generic_space;
96  decode_space(space, thread, actual_param_op, mem,
97  addr); // figure out which space
98  if (arg == 0) {
99  unsigned len = 0;
100  char b = 0;
101  do { // figure out length
102  mem->read(addr + len, 1, &b);
103  len++;
104  } while (b);
105  fmtstr = (char *)malloc(len + 64);
106  for (int i = 0; i < len; i++) mem->read(addr + i, 1, fmtstr + i);
107  // mem->read(addr,len,fmtstr);
108  } else {
109  unsigned len = thread->get_finfo()->local_mem_framesize();
110  arg_list = (char *)malloc(len + 64);
111  for (int i = 0; i < len; i++) mem->read(addr + i, 1, arg_list + i);
112  // mem->read(addr,len,arg_list);
113  }
114  }
115  my_cuda_printf(fmtstr, arg_list);
116  free(fmtstr);
117  free(arg_list);
118 }
